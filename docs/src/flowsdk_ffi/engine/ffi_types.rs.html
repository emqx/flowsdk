41#[cfg_attr(feature = "uniffi-bindings", derive(uniffi::Enum, serde::Serialize))]
42#[derive(Clone)]
43pub enum MqttEventFFI {
44    Connected(ConnectionResultFFI),
45    Disconnected {
46        reason_code: Option<u8>,
47    },
48    MessageReceived(MqttMessageFFI),
49    Published(PublishResultFFI),
50    Subscribed(SubscribeResultFFI),
51    Unsubscribed(UnsubscribeResultFFI),
52    PingResponse {
53        success: bool,
54    },
55    Error {
56        message: String,
57    },
58    StreamClosed {
59        stream_id: u64,
60        reason: String,
61        by_peer: bool,
62    },
63    StreamReset {
64        stream_id: u64,
65        error_code: u64,
66    },
67    StreamStopped {
68        stream_id: u64,
69        error_code: u64,
70    },
71    ReconnectNeeded,
72    ReconnectScheduled {
73        attempt: u32,
74        delay_ms: u64,
75    },
76}
