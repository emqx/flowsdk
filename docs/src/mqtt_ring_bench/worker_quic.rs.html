19pub fn run_quic_worker(
20    worker_id: usize,
21    conn_range: std::ops::Range<usize>,
22    config: Arc<BenchConfig>,
23    stats: Arc<BenchStats>,
24    crypto: Arc<rustls::ClientConfig>,
25) -> WorkerResult {
26    let num_conns = conn_range.len();
27    if num_conns == 0 {
28        return WorkerResult {
29            latency_samples: Vec::new(),
30        };
31    }
32
33    let ring_size = 4096u32.min((num_conns as u32 * 3).next_power_of_two().max(64));
34    let mut ring: IoUring = IoUring::builder()
35        .build(ring_size)
36        .unwrap_or_else(|e| panic!("worker {}: failed to create io_uring: {}", worker_id, e));
37
38    let mut conns: Slab<Box<QuicConnection>> = Slab::with_capacity(num_conns);
39
40    let server_name = config
41        .server_name
42        .as_deref()
43        .unwrap_or(&config.host)
44        .to_string();
45
46    let budget_per_sec = config.connect_rate / config.workers.max(1);
47    let mut connect_budget: f64 = 0.0;
48    let mut last_budget_tick = Instant::now();
49    let mut pending_create: Vec<usize> = conn_range.collect();
50    pending_create.reverse();
51
52    let mut done_count: usize = 0;
53    let mut ifaddr_idx: usize = 0;
54
55    loop {
56        if stats.stopped.load(Ordering::Relaxed) {
57            break;
58        }
59
60        let now = Instant::now();
61
62        // Replenish connect budget
63        let elapsed_secs = now.duration_since(last_budget_tick).as_secs_f64();
64        if elapsed_secs >= 0.01 {
65            connect_budget += budget_per_sec as f64 * elapsed_secs;
66            if connect_budget > pending_create.len() as f64 {
67                connect_budget = pending_create.len() as f64;
68            }
69            last_budget_tick = now;
70        }
71
72        // Create UDP sockets, initiate QUIC handshake
73        while connect_budget >= 1.0 {
74            if let Some(client_idx) = pending_create.pop() {
75                let bind_ip = if config.ifaddrs.is_empty() {
76                    None
77                } else {
78                    let ip = config.ifaddrs[ifaddr_idx % config.ifaddrs.len()];
79                    ifaddr_idx += 1;
80                    Some(ip)
81                };
82                match create_udp_socket(&config, bind_ip) {
83                    Ok(fd) => {
84                        // "connect" the UDP socket so Send/Recv work without sendto/recvfrom
85                        let ret = udp_connect(fd, &config.addr);
86                        if ret < 0 {
87                            let e = std::io::Error::last_os_error();
88                            eprintln!(
89                                "worker {}: UDP connect failed for client {}: {}",
90                                worker_id, client_idx, e
91                            );
92                            unsafe {
93                                libc::close(fd);
94                            }
95                            stats.errors.fetch_add(1, Ordering::Relaxed);
96                            stats.clients_done.fetch_add(1, Ordering::Relaxed);
97                            done_count += 1;
98                            connect_budget -= 1.0;
99                            continue;
100                        }
101
102                        match QuicConnection::new(fd, client_idx, &config, config.addr) {
103                            Ok(mut conn) => {
104                                let crypto_clone = (*crypto).clone();
105                                match conn.initiate_quic_connect(crypto_clone, &server_name, now) {
106                                    Ok(()) => {
107                                        let key = conns.insert(Box::new(conn));
108                                        // Submit initial handshake datagrams + recv
109                                        submit_front_send(&mut ring, key, &mut conns[key]);
110                                        submit_recv(&mut ring, key, &mut conns[key]);
111                                    }
112                                    Err(e) => {
113                                        eprintln!(
114                                            "worker {}: QUIC connect failed for client {}: {}",
115                                            worker_id, client_idx, e
116                                        );
117                                        unsafe {
118                                            libc::close(fd);
119                                        }
120                                        stats.errors.fetch_add(1, Ordering::Relaxed);
121                                        stats.clients_done.fetch_add(1, Ordering::Relaxed);
122                                        done_count += 1;
123                                    }
124                                }
125                            }
126                            Err(e) => {
127                                eprintln!(
128                                    "worker {}: QuicConnection::new failed for client {}: {}",
129                                    worker_id, client_idx, e
130                                );
131                                unsafe {
132                                    libc::close(fd);
133                                }
134                                stats.errors.fetch_add(1, Ordering::Relaxed);
135                                stats.clients_done.fetch_add(1, Ordering::Relaxed);
136                                done_count += 1;
137                            }
138                        }
139                        connect_budget -= 1.0;
140                    }
141                    Err(e) => {
142                        eprintln!("worker {}: UDP socket creation failed: {}", worker_id, e);
143                        stats.errors.fetch_add(1, Ordering::Relaxed);
144                        stats.clients_done.fetch_add(1, Ordering::Relaxed);
145                        done_count += 1;
146                        connect_budget -= 1.0;
147                    }
148                }
149            } else {
150                break;
151            }
152        }
153
154        // Wait up to 10ms for completions (QUIC needs frequent ticking)
155        let ts = Timespec::new().nsec(10_000_000); // 10ms
156        let args = SubmitArgs::new().timespec(&ts);
157        let wait_count = if conns.is_empty() { 0 } else { 1 };
158        match ring.submitter().submit_with_args(wait_count, &args) {
159            Ok(_) => {}
160            Err(ref e) if e.raw_os_error() == Some(libc::EINTR) => continue,
161            Err(ref e) if e.raw_os_error() == Some(libc::EBUSY) => continue,
162            Err(ref e) if e.raw_os_error() == Some(libc::ETIME) => {} // timeout, continue
163            Err(e) => {
164                eprintln!("worker {}: io_uring submit error: {}", worker_id, e);
165                break;
166            }
167        }
168
169        // Process completions
170        let now = Instant::now();
171        let cqes: Vec<_> = ring.completion().collect();
172        for cqe in cqes {
173            let (conn_key, op) = decode_user_data(cqe.user_data());
174            let result = cqe.result();
175
176            if !conns.contains(conn_key) {
177                continue;
178            }
179
180            match op {
181                OP_SEND => {
182                    handle_send_complete(
183                        &mut conns,
184                        conn_key,
185                        result,
186                        &mut ring,
187                        &stats,
188                        &mut done_count,
189                    );
190                }
191                OP_RECV => {
192                    handle_recv_complete(
193                        &mut conns,
194                        conn_key,
195                        result,
196                        &mut ring,
197                        &config,
198                        &stats,
199                        &mut done_count,
200                    );
201                }
202                _ => {}
203            }
204        }
205
206        // Drive QUIC + MQTT state machines. The io_uring wait timeout (10ms)
207        // provides the cadence; quinn-proto's loss detection and pacing need
208        // ticks at roughly that rate regardless of CQE arrivals.
209        {
210            let keys: Vec<usize> = conns
211                .iter()
212                .filter(|(_, c)| c.state != QuicConnState::Done && c.state != QuicConnState::Failed)
213                .map(|(k, _)| k)
214                .collect();
215            for key in keys {
216                let conn = &mut conns[key];
217
218                // Drive QUIC state machine
219                let events = conn.handle_tick(now);
220                if !events.is_empty() {
221                    let outcome = conn.process_events(events);
222                    if outcome.connected {
223                        stats.clients_connected.fetch_add(1, Ordering::Relaxed);
224                    }
225                    if outcome.acked > 0 {
226                        stats
227                            .messages_acked
228                            .fetch_add(outcome.acked, Ordering::Relaxed);
229                    }
230                    if outcome.error {
231                        stats.errors.fetch_add(1, Ordering::Relaxed);
232                    }
233                }
234
235                // Try publish
236                if conn.state == QuicConnState::Publishing {
237                    if conn.try_publish(&config, now) {
238                        stats.messages_sent.fetch_add(1, Ordering::Relaxed);
239                    }
240                    conn.check_publish_complete();
241                }
242
243                // Check drain
244                if conn.state == QuicConnState::Draining {
245                    conn.check_drain_complete(now);
246                }
247
248                // Check if disconnecting is done (no more sends pending)
249                if conn.state == QuicConnState::Disconnecting
250                    && !conn.has_pending_send()
251                    && !conn.send_pending
252                {
253                    conn.state = QuicConnState::Done;
254                    stats.clients_done.fetch_add(1, Ordering::Relaxed);
255                    done_count += 1;
256                }
257
258                // Submit any new outgoing datagrams
259                if conn.has_pending_send() && !conn.send_pending {
260                    submit_front_send(&mut ring, key, conn);
261                }
262            }
263        }
264
265        if done_count >= num_conns && pending_create.is_empty() {
266            break;
267        }
268    }
269
270    // Collect latency samples
271    let mut all_samples = Vec::new();
272    for (_, conn) in conns.iter_mut() {
273        all_samples.append(&mut conn.latency_samples);
274    }
275
276    // Close fds
277    for (_, conn) in conns.iter() {
278        unsafe {
279            libc::close(conn.fd);
280        }
281    }
282
283    WorkerResult {
284        latency_samples: all_samples,
285    }
286}

288fn create_udp_socket(
289    config: &BenchConfig,
290    bind_addr: Option<std::net::IpAddr>,
291) -> std::io::Result<RawFd> {
292    let domain = if config.addr.is_ipv4() {
293        Domain::IPV4
294    } else {
295        Domain::IPV6
296    };
297    let socket = Socket::new(domain, Type::DGRAM, None)?;
298    socket.set_nonblocking(true)?;
299    socket.set_recv_buffer_size(config.socket_buf)?;
300    socket.set_send_buffer_size(config.socket_buf)?;
301
302    let bind_sa = match bind_addr {
303        Some(ip) => SockAddr::from(std::net::SocketAddr::new(ip, 0)),
304        None => {
305            // Bind to 0.0.0.0:0 so the OS assigns an ephemeral port
306            if config.addr.is_ipv4() {
307                SockAddr::from(std::net::SocketAddr::new(
308                    std::net::IpAddr::V4(std::net::Ipv4Addr::UNSPECIFIED),
309                    0,
310                ))
311            } else {
312                SockAddr::from(std::net::SocketAddr::new(
313                    std::net::IpAddr::V6(std::net::Ipv6Addr::UNSPECIFIED),
314                    0,
315                ))
316            }
317        }
318    };
319    socket.bind(&bind_sa)?;
320
321    Ok(socket.into_raw_fd())
322}

324/// Connect a UDP socket to the given address using libc::connect.
325/// This sets the default peer so Send/Recv work like on a connected socket.
326fn udp_connect(fd: RawFd, addr: &std::net::SocketAddr) -> i32 {
327    let sa = SockAddr::from(*addr);
328    unsafe { libc::connect(fd, sa.as_ptr(), sa.len() as libc::socklen_t) }
329}

331fn handle_send_complete(
332    conns: &mut Slab<Box<QuicConnection>>,
333    key: usize,
334    result: i32,
335    ring: &mut IoUring,
336    stats: &BenchStats,
337    done_count: &mut usize,
338) {
339    let conn = &mut conns[key];
340    conn.send_pending = false;
341
342    if result < 0 {
343        let err_code = -result;
344        eprintln!(
345            "connection {}: UDP send failed: {}",
346            conn.client_index,
347            std::io::Error::from_raw_os_error(err_code)
348        );
349        mark_failed(conn, stats, done_count);
350        return;
351    }
352
353    // Datagram sent successfully; pop it from the queue
354    conn.pop_front_datagram();
355
356    // If more datagrams in queue, submit next send
357    if conn.has_pending_send() {
358        submit_front_send(ring, key, conn);
359    }
360}

362fn handle_recv_complete(
363    conns: &mut Slab<Box<QuicConnection>>,
364    key: usize,
365    result: i32,
366    ring: &mut IoUring,
367    config: &BenchConfig,
368    stats: &BenchStats,
369    done_count: &mut usize,
370) {
371    let conn = &mut conns[key];
372    conn.recv_pending = false;
373
374    if result <= 0 {
375        if result < 0 {
376            let err_code = -result;
377            // ECONNREFUSED is common for UDP when the remote port is closed
378            if err_code != libc::ECONNREFUSED {
379                eprintln!(
380                    "connection {}: UDP recv error: {}",
381                    conn.client_index,
382                    std::io::Error::from_raw_os_error(err_code)
383                );
384            }
385        }
386        if conn.state != QuicConnState::Done && conn.state != QuicConnState::Disconnecting {
387            mark_failed(conn, stats, done_count);
388        }
389        return;
390    }
391
392    let nbytes = result as usize;
393    let now = Instant::now();
394
395    // Feed datagram to QUIC engine
396    let data: Vec<u8> = conn.recv_buf[..nbytes].to_vec();
397    let events = conn.handle_datagram(&data, now);
398
399    if !events.is_empty() {
400        let outcome = conn.process_events(events);
401        if outcome.connected {
402            stats.clients_connected.fetch_add(1, Ordering::Relaxed);
403        }
404        if outcome.acked > 0 {
405            stats
406                .messages_acked
407                .fetch_add(outcome.acked, Ordering::Relaxed);
408        }
409        if outcome.error {
410            stats.errors.fetch_add(1, Ordering::Relaxed);
411        }
412    }
413
414    // Try to publish after receiving (may have transitioned to Publishing)
415    if conn.state == QuicConnState::Publishing {
416        if conn.try_publish(config, now) {
417            stats.messages_sent.fetch_add(1, Ordering::Relaxed);
418        }
419        conn.check_publish_complete();
420    }
421
422    if conn.state == QuicConnState::Draining {
423        conn.check_drain_complete(now);
424    }
425
426    // Submit outgoing datagrams
427    if conn.has_pending_send() && !conn.send_pending {
428        submit_front_send(ring, key, conn);
429    }
430
431    // Continue receiving (unless done/failed)
432    if conn.state != QuicConnState::Done && conn.state != QuicConnState::Failed {
433        submit_recv(ring, key, conn);
434    }
435}

437fn mark_failed(conn: &mut QuicConnection, stats: &BenchStats, done_count: &mut usize) {
438    if conn.state != QuicConnState::Failed && conn.state != QuicConnState::Done {
439        conn.state = QuicConnState::Failed;
440        stats.errors.fetch_add(1, Ordering::Relaxed);
441        stats.clients_done.fetch_add(1, Ordering::Relaxed);
442        *done_count += 1;
443    }
444}

446fn submit_front_send(ring: &mut IoUring, key: usize, conn: &mut QuicConnection) {
447    if conn.send_pending {
448        return;
449    }
450    let datagram = match conn.front_send_datagram() {
451        Some(d) => d,
452        None => return,
453    };
454    let ptr = datagram.as_ptr();
455    let len = datagram.len() as u32;
456
457    let send_e = opcode::Send::new(Fd(conn.fd), ptr, len)
458        .build()
459        .user_data(encode_user_data(key, OP_SEND));
460
461    conn.send_pending = true;
462    unsafe {
463        if ring.submission().push(&send_e).is_err() {
464            let _ = ring.submitter().submit();
465            let _ = ring.submission().push(&send_e);
466        }
467    }
468}
469
470fn submit_recv(ring: &mut IoUring, key: usize, conn: &mut QuicConnection) {
471    if conn.recv_pending {
472        return;
473    }
474    let buf_ptr = conn.recv_buf.as_mut_ptr();
475    let buf_len = conn.recv_buf.len() as u32;
476
477    let recv_e = opcode::Recv::new(Fd(conn.fd), buf_ptr, buf_len)
478        .build()
479        .user_data(encode_user_data(key, OP_RECV));
480
481    conn.recv_pending = true;
482    unsafe {
483        if ring.submission().push(&recv_e).is_err() {
484            let _ = ring.submitter().submit();
485            let _ = ring.submission().push(&recv_e);
486        }
487    }
