3use std::sync::atomic::{AtomicBool, AtomicU64, Ordering};
4use std::time::{Duration, Instant};
5
6pub struct BenchStats {
7    pub messages_sent: AtomicU64,
8    pub messages_acked: AtomicU64,
9    pub errors: AtomicU64,
10    pub clients_connected: AtomicU64,
11    pub clients_done: AtomicU64,
12    pub start_time: Instant,
13    pub stopped: AtomicBool,
14}
15
16impl BenchStats {
17    pub fn new() -> Self {
18        Self {
19            messages_sent: AtomicU64::new(0),
20            messages_acked: AtomicU64::new(0),
21            errors: AtomicU64::new(0),
22            clients_connected: AtomicU64::new(0),
23            clients_done: AtomicU64::new(0),
24            start_time: Instant::now(),
25            stopped: AtomicBool::new(false),
26        }
27    }
28
29    pub fn snapshot(&self) -> StatsSnapshot {
30        StatsSnapshot {
31            sent: self.messages_sent.load(Ordering::Relaxed),
32            acked: self.messages_acked.load(Ordering::Relaxed),
33            errors: self.errors.load(Ordering::Relaxed),
34            connected: self.clients_connected.load(Ordering::Relaxed),
35            done: self.clients_done.load(Ordering::Relaxed),
36            elapsed: self.start_time.elapsed(),
37        }
38    }
39}

50pub fn merge_latency_samples(worker_samples: Vec<Vec<Duration>>) -> Vec<Duration> {
51    let total: usize = worker_samples.iter().map(|v| v.len()).sum();
52    let mut merged = Vec::with_capacity(total);
53    for mut samples in worker_samples {
54        merged.append(&mut samples);
55    }
56    merged.sort_unstable();
57    merged
58}

60pub fn percentile(sorted: &[Duration], pct: f64) -> Duration {
61    if sorted.is_empty() {
62        return Duration::ZERO;
63    }
64    let idx = ((sorted.len() as f64 - 1.0) * pct / 100.0) as usize;
65    sorted[idx.min(sorted.len() - 1)]
66}

68pub fn print_final_summary(
69    config: &crate::config::BenchConfig,
70    snap: &StatsSnapshot,
71    latency_samples: &[Duration],
72) {
73    let duration_secs = snap.elapsed.as_secs_f64();
74    let throughput = if duration_secs > 0.0 {
75        snap.sent as f64 / duration_secs
76    } else {
77        0.0
78    };
79
80    let sep = "=".repeat(60);
81    let dash = "-".repeat(60);
82    println!("\n{sep}");
83    println!("{:^60}", "MQTT Bench Results");
84    println!("{sep}");
85    let transport = if config.quic { "QUIC" } else { "TCP" };
86    println!(
87        "  Broker:          {}:{} ({})",
88        config.host, config.port, transport
89    );
90    println!("  MQTT Version:    {}", config.mqtt_version);
91    println!("  Clients:         {}", config.clients);
92    println!("  Workers:         {}", config.workers);
93    println!("  QoS:             {}", config.qos);
94    println!("  Payload Size:    {} bytes", config.payload_size);
95    println!("  Messages/Client: {}", config.messages);
96    println!("  Socket Buffers:  {} bytes", config.socket_buf);
97    println!("  Parser Buffer:   {} bytes", config.parser_buf);
98    println!("{dash}");
99    println!("  Total Sent:      {}", snap.sent);
100    if config.qos > 0 {
101        println!("  Total Acked:     {}", snap.acked);
102    }
103    println!("  Errors:          {}", snap.errors);
104    println!("  Duration:        {:.2} s", duration_secs);
105    println!("  Throughput:      {:.2} msg/s", throughput);
106
107    if !latency_samples.is_empty() {
108        println!("{dash}");
109        let label = if config.qos == 0 {
110            "Latency (enqueue time)"
111        } else {
112            "Latency (send -> ACK)"
113        };
114        println!("  {}:", label);
115        let avg: Duration = latency_samples.iter().sum::<Duration>() / latency_samples.len() as u32;
116        println!("    Min:           {:.2} ms", ms(latency_samples[0]));
117        println!(
118            "    Max:           {:.2} ms",
119            ms(*latency_samples.last().unwrap())
120        );
121        println!("    Avg:           {:.2} ms", ms(avg));
122        println!(
123            "    P50:           {:.2} ms",
124            ms(percentile(latency_samples, 50.0))
125        );
126        println!(
127            "    P95:           {:.2} ms",
128            ms(percentile(latency_samples, 95.0))
129        );
130        println!(
131            "    P99:           {:.2} ms",
132            ms(percentile(latency_samples, 99.0))
133        );
134        println!("    Samples:       {}", latency_samples.len());
135    }
136    println!("{sep}");
137}
