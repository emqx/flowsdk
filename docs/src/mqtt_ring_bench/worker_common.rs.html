3use io_uring::opcode;
4use io_uring::types::{SubmitArgs, Timespec};
5use io_uring::IoUring;
6use std::collections::HashSet;
7use std::io;
8use std::time::Duration;

10pub const OP_CONNECT: u64 = 0;
11pub const OP_SEND: u64 = 1;
12pub const OP_RECV: u64 = 2;
13
14const CANCEL_USER_DATA_BIT: u64 = 1 << 63;
15const SHUTDOWN_WAIT: Duration = Duration::from_millis(100);
16
17pub fn encode_user_data(conn_key: usize, op: u64) -> u64 {
18    (op << 48) | (conn_key as u64)
19}
20
21pub fn decode_user_data(data: u64) -> (usize, u64) {
22    let conn_key = (data & 0x0000_FFFF_FFFF_FFFF) as usize;
23    let op = data >> 48;
24    (conn_key, op)
25}
26
27pub fn pending_user_data(
28    conn_key: usize,
29    connect_pending: bool,
30    send_pending: bool,
31    recv_pending: bool,
32) -> impl Iterator<Item = u64> {
33    [
34        connect_pending.then(|| encode_user_data(conn_key, OP_CONNECT)),
35        send_pending.then(|| encode_user_data(conn_key, OP_SEND)),
36        recv_pending.then(|| encode_user_data(conn_key, OP_RECV)),
37    ]
38    .into_iter()
39    .flatten()
40}

42/// Cancel and reap every operation before its socket and backing buffers are dropped.
43pub fn cancel_pending_operations(ring: &mut IoUring, pending: Vec<u64>) -> io::Result<()> {
44    let mut shutdown = ShutdownCompletions::new(&pending);
45
46    for target in pending {
47        let cancel_data = cancel_user_data(target);
48        let cancel = opcode::AsyncCancel::new(target)
49            .build()
50            .user_data(cancel_data);
51
52        loop {
53            if unsafe { ring.submission().push(&cancel) }.is_ok() {
54                shutdown.cancellations.insert(cancel_data);
55                break;
56            }
57            submit_shutdown_batch(ring, &mut shutdown)?;
58        }
59    }
60
61    submit_shutdown_batch(ring, &mut shutdown)?;
62    while !shutdown.is_empty() {
63        drain_shutdown_completions(ring, &mut shutdown);
64        if shutdown.is_empty() {
65            break;
66        }
67
68        let ts = Timespec::new()
69            .sec(SHUTDOWN_WAIT.as_secs())
70            .nsec(SHUTDOWN_WAIT.subsec_nanos());
71        let args = SubmitArgs::new().timespec(&ts);
72        match ring.submitter().submit_with_args(1, &args) {
73            Ok(_) => {}
74            Err(ref error)
75                if matches!(
76                    error.raw_os_error(),
77                    Some(libc::EINTR | libc::EBUSY | libc::ETIME)
78                ) => {}
79            Err(error) => return Err(error),
80        }
81    }
82
83    Ok(())
84}
85
86fn submit_shutdown_batch(ring: &mut IoUring, shutdown: &mut ShutdownCompletions) -> io::Result<()> {
87    loop {
88        match ring.submitter().submit() {
89            Ok(_) => break,
90            Err(ref error) if matches!(error.raw_os_error(), Some(libc::EINTR | libc::EBUSY)) => {
91                drain_shutdown_completions(ring, shutdown);
92            }
93            Err(error) => return Err(error),
94        }
95    }
96    drain_shutdown_completions(ring, shutdown);
97    Ok(())
98}
99
100fn drain_shutdown_completions(ring: &mut IoUring, shutdown: &mut ShutdownCompletions) {
101    for cqe in ring.completion() {
102        shutdown.record(cqe.user_data(), cqe.result());
103    }
104}
105
106struct ShutdownCompletions {
107    operations: HashSet<u64>,
108    cancellations: HashSet<u64>,
109}
110
111impl ShutdownCompletions {
112    fn new(pending: &[u64]) -> Self {
113        Self {
114            operations: pending.iter().copied().collect(),
115            cancellations: HashSet::with_capacity(pending.len()),
116        }
117    }
118
119    fn is_empty(&self) -> bool {
120        self.operations.is_empty() && self.cancellations.is_empty()
121    }
122
123    fn record(&mut self, user_data: u64, result: i32) {
124        if let Some(target) = cancel_target(user_data) {
125            self.cancellations.remove(&user_data);
126            // ENOENT means no request with this user_data remains in the ring.
127            if result == -libc::ENOENT {
128                self.operations.remove(&target);
129            }
130        } else {
131            self.operations.remove(&user_data);
132        }
133    }
134}
135
136fn cancel_user_data(target: u64) -> u64 {
137    debug_assert_eq!(target & CANCEL_USER_DATA_BIT, 0);
138    target | CANCEL_USER_DATA_BIT
139}
140
141fn cancel_target(user_data: u64) -> Option<u64> {
142    (user_data & CANCEL_USER_DATA_BIT != 0).then_some(user_data & !CANCEL_USER_DATA_BIT)
143}

149#[cfg(test)]
150mod tests {
151    use super::*;
152
153    #[test]
154    fn pending_user_data_contains_only_active_operations() {
155        assert_eq!(
156            pending_user_data(42, true, false, true).collect::<Vec<_>>(),
157            vec![
158                encode_user_data(42, OP_CONNECT),
159                encode_user_data(42, OP_RECV),
160            ]
161        );
162    }
163
164    #[test]
165    fn cancellation_tag_round_trips_target() {
166        let target = encode_user_data(123, OP_SEND);
167        let tagged = cancel_user_data(target);
168
169        assert_eq!(cancel_target(tagged), Some(target));
170        assert_eq!(cancel_target(target), None);
171    }
172
173    #[test]
174    fn shutdown_waits_for_cancel_after_operation_completion() {
175        let target = encode_user_data(123, OP_RECV);
176        let cancel = cancel_user_data(target);
177        let mut shutdown = ShutdownCompletions::new(&[target]);
178        shutdown.cancellations.insert(cancel);
179
180        shutdown.record(target, -libc::ECANCELED);
181        assert!(!shutdown.is_empty());
182
183        shutdown.record(cancel, 0);
184        assert!(shutdown.is_empty());
185    }
186
187    #[test]
188    fn shutdown_waits_for_operation_after_cancel_completion() {
189        let target = encode_user_data(456, OP_SEND);
190        let cancel = cancel_user_data(target);
191        let mut shutdown = ShutdownCompletions::new(&[target]);
192        shutdown.cancellations.insert(cancel);
193
194        shutdown.record(cancel, 0);
195        assert!(!shutdown.is_empty());
196
197        shutdown.record(target, -libc::ECANCELED);
198        assert!(shutdown.is_empty());
199    }
200
201    #[test]
202    fn cancel_enoent_completes_missing_operation() {
203        let target = encode_user_data(789, OP_CONNECT);
204        let cancel = cancel_user_data(target);
205        let mut shutdown = ShutdownCompletions::new(&[target]);
206        shutdown.cancellations.insert(cancel);
207
208        shutdown.record(cancel, -libc::ENOENT);
209
210        assert!(shutdown.is_empty());
211    }
