21/// Each Connection is Box-allocated so its buffers have stable addresses
22/// for io_uring SQEs (Slab<T> may relocate entries on grow).
23pub fn run_worker(
24    worker_id: usize,
25    conn_range: std::ops::Range<usize>,
26    config: Arc<BenchConfig>,
27    stats: Arc<BenchStats>,
28) -> WorkerResult {
29    let num_conns = conn_range.len();
30    if num_conns == 0 {
31        return WorkerResult {
32            latency_samples: Vec::new(),
33        };
34    }
35
36    let ring_size = 4096u32.min((num_conns as u32 * 3).next_power_of_two().max(64));
37    let mut ring: IoUring = IoUring::builder()
38        .build(ring_size)
39        .unwrap_or_else(|e| panic!("worker {}: failed to create io_uring: {}", worker_id, e));
40
41    let mut conns: Slab<Box<Connection>> = Slab::with_capacity(num_conns);
42
43    // We need SockAddr to stay alive for the duration of connect SQEs.
44    // Keep one per in-flight connect. Once connected, it can be dropped.
45    let sock_addr = SockAddr::from(config.addr);
46    // Pin the SockAddr in a Box so its address is stable for io_uring.
47    let sock_addr_box = Box::new(sock_addr);
48    let sa_ptr = sock_addr_box.as_ref().as_ptr();
49    let sa_len = sock_addr_box.as_ref().len();
50
51    let budget_per_sec = config.connect_rate / config.workers.max(1);
52    let mut connect_budget: f64 = 0.0;
53    let mut last_budget_tick = Instant::now();
54    let mut pending_create: Vec<usize> = conn_range.collect();
55    pending_create.reverse();
56
57    let mut done_count: usize = 0;
58    let mut ifaddr_idx: usize = 0;
59    // Upper bound on idle wait — keeps Ctrl+C / shutdown responsive even when
60    // no protocol timer is scheduled.
61    let max_idle_wait = Duration::from_secs(1);
62
63    loop {
64        if stats.stopped.load(Ordering::Relaxed) {
65            break;
66        }
67
68        let now = Instant::now();
69
70        // Replenish connect budget using fractional accumulation
71        let elapsed_secs = now.duration_since(last_budget_tick).as_secs_f64();
72        if elapsed_secs >= 0.01 {
73            connect_budget += budget_per_sec as f64 * elapsed_secs;
74            if connect_budget > pending_create.len() as f64 {
75                connect_budget = pending_create.len() as f64;
76            }
77            last_budget_tick = now;
78        }
79
80        // Create sockets and submit connect SQEs
81        while connect_budget >= 1.0 {
82            if let Some(client_idx) = pending_create.pop() {
83                let bind_ip = if config.ifaddrs.is_empty() {
84                    None
85                } else {
86                    let ip = config.ifaddrs[ifaddr_idx % config.ifaddrs.len()];
87                    ifaddr_idx += 1;
88                    Some(ip)
89                };
90                match create_socket(&config, bind_ip) {
91                    Ok(fd) => {
92                        let conn = Box::new(Connection::new(fd, client_idx, &config));
93                        let key = conns.insert(conn);
94
95                        let connect_e = opcode::Connect::new(Fd(fd), sa_ptr as *const _, sa_len)
96                            .build()
97                            .user_data(encode_user_data(key, OP_CONNECT));
98
99                        unsafe {
100                            if ring.submission().push(&connect_e).is_err() {
101                                // SQ full, submit and retry
102                                let _ = ring.submitter().submit();
103                                let _ = ring.submission().push(&connect_e);
104                            }
105                        }
106                        connect_budget -= 1.0;
107                    }
108                    Err(e) => {
109                        eprintln!("worker {}: socket creation failed: {}", worker_id, e);
110                        stats.errors.fetch_add(1, Ordering::Relaxed);
111                        stats.clients_done.fetch_add(1, Ordering::Relaxed);
112                        done_count += 1;
113                        connect_budget -= 1.0;
114                    }
115                }
116            } else {
117                break;
118            }
119        }
120
121        // Wait for the soonest of: next protocol deadline, a CQE arrival, or
122        // `max_idle_wait` (keeps shutdown responsive when nothing is scheduled).
123        let min_deadline = conns
124            .iter()
125            .filter(|(_, c)| {
126                c.state != ConnState::Done
127                    && c.state != ConnState::Failed
128                    && c.state != ConnState::TcpConnecting
129            })
130            .filter_map(|(_, c)| c.next_tick_at())
131            .min();
132        let wait_dur = match min_deadline {
133            Some(d) => d.saturating_duration_since(now).min(max_idle_wait),
134            None => max_idle_wait,
135        };
136        let wait_count = if conns.is_empty() || wait_dur.is_zero() {
137            0
138        } else {
139            1
140        };
141        let ts = Timespec::new()
142            .sec(wait_dur.as_secs())
143            .nsec(wait_dur.subsec_nanos());
144        let args = SubmitArgs::new().timespec(&ts);
145        match ring.submitter().submit_with_args(wait_count, &args) {
146            Ok(_) => {}
147            Err(ref e) if e.raw_os_error() == Some(libc::EINTR) => continue,
148            Err(ref e) if e.raw_os_error() == Some(libc::EBUSY) => continue,
149            Err(ref e) if e.raw_os_error() == Some(libc::ETIME) => {} // deadline hit, fall through to tick
150            Err(e) => {
151                eprintln!("worker {}: io_uring submit error: {}", worker_id, e);
152                break;
153            }
154        }
155
156        // Process completions
157        let cqes: Vec<_> = ring.completion().collect();
158        for cqe in cqes {
159            let (conn_key, op) = decode_user_data(cqe.user_data());
160            let result = cqe.result();
161
162            if !conns.contains(conn_key) {
163                continue;
164            }
165
166            match op {
167                OP_CONNECT => {
168                    handle_connect_complete(&mut conns, conn_key, result, &mut ring, &stats);
169                }
170                OP_SEND => {
171                    handle_send_complete(
172                        &mut conns,
173                        conn_key,
174                        result,
175                        &mut ring,
176                        &config,
177                        &stats,
178                        &mut done_count,
179                    );
180                }
181                OP_RECV => {
182                    handle_recv_complete(
183                        &mut conns,
184                        conn_key,
185                        result,
186                        &mut ring,
187                        &config,
188                        &stats,
189                        &mut done_count,
190                    );
191                }
192                _ => {}
193            }
194        }
195
196        // Try to publish on ready connections
197        let keys: Vec<usize> = conns
198            .iter()
199            .filter(|(_, c)| c.state == ConnState::Publishing && !c.send_pending)
200            .map(|(k, _)| k)
201            .collect();
202        for key in keys {
203            let conn = &mut conns[key];
204            if conn.try_publish(&config) {
205                stats.messages_sent.fetch_add(1, Ordering::Relaxed);
206                submit_send(&mut ring, key, conn);
207            }
208            conn.check_publish_complete();
209            if conn.state == ConnState::Draining
210                && !conn.send_pending
211                && conn.check_drain_complete()
212            {
213                submit_send(&mut ring, key, conn);
214            }
215        }
216
217        // Drive protocol timers. Connection::handle_tick is a cheap no-op if
218        // the per-connection deadline hasn't been reached, so iterating every
219        // loop pass is fine — and only happens when we either got a CQE or
220        // crossed the global min_deadline computed above.
221        let tick_keys: Vec<usize> = conns
222            .iter()
223            .filter(|(_, c)| {
224                c.state != ConnState::Done
225                    && c.state != ConnState::Failed
226                    && c.state != ConnState::TcpConnecting
227            })
228            .map(|(k, _)| k)
229            .collect();
230        for key in tick_keys {
231            let conn = &mut conns[key];
232            let events = conn.handle_tick();
233            if !events.is_empty() {
234                let outcome = conn.process_events(events);
235                if outcome.error {
236                    stats.errors.fetch_add(1, Ordering::Relaxed);
237                }
238                if outcome.acked > 0 {
239                    stats
240                        .messages_acked
241                        .fetch_add(outcome.acked, Ordering::Relaxed);
242                }
243            }
244            // PINGREQ / retransmits produce outgoing bytes without any event.
245            if conn.has_pending_send() && !conn.send_pending {
246                submit_send(&mut ring, key, conn);
247            }
248        }
249
250        if done_count >= num_conns && pending_create.is_empty() {
251            break;
252        }
253    }
254
255    // Collect latency samples
256    let mut all_samples = Vec::new();
257    for (_, conn) in conns.iter_mut() {
258        all_samples.append(&mut conn.latency_samples);
259    }
260
261    // Close fds
262    for (_, conn) in conns.iter() {
263        unsafe {
264            libc::close(conn.fd);
265        }
266    }
267
268    drop(sock_addr_box);
269
270    WorkerResult {
271        latency_samples: all_samples,
272    }
273}

275fn create_socket(
276    config: &BenchConfig,
277    bind_addr: Option<std::net::IpAddr>,
278) -> std::io::Result<RawFd> {
279    let domain = if config.addr.is_ipv4() {
280        Domain::IPV4
281    } else {
282        Domain::IPV6
283    };
284    let socket = Socket::new(domain, Type::STREAM, Some(Protocol::TCP))?;
285    socket.set_nonblocking(true)?;
286    socket.set_nodelay(true)?;
287    socket.set_recv_buffer_size(config.socket_buf)?;
288    socket.set_send_buffer_size(config.socket_buf)?;
289
290    // Bind to a specific source IP; port 0 lets the OS pick an ephemeral port
291    if let Some(ip) = bind_addr {
292        let bind_sa = SockAddr::from(std::net::SocketAddr::new(ip, 0));
293        socket.bind(&bind_sa)?;
294    }
295
296    Ok(socket.into_raw_fd())
297}

299fn handle_connect_complete(
300    conns: &mut Slab<Box<Connection>>,
301    key: usize,
302    result: i32,
303    ring: &mut IoUring,
304    stats: &BenchStats,
305) {
306    let conn = &mut conns[key];
307    if result < 0 {
308        eprintln!(
309            "connection {}: TCP connect failed: {}",
310            conn.client_index,
311            std::io::Error::from_raw_os_error(-result)
312        );
313        conn.state = ConnState::Failed;
314        stats.errors.fetch_add(1, Ordering::Relaxed);
315        stats.clients_done.fetch_add(1, Ordering::Relaxed);
316        return;
317    }
318
319    stats.clients_connected.fetch_add(1, Ordering::Relaxed);
320    conn.initiate_mqtt_connect();
321    submit_send(ring, key, conn);
322    submit_recv(ring, key, conn);
323}

325fn handle_send_complete(
326    conns: &mut Slab<Box<Connection>>,
327    key: usize,
328    result: i32,
329    ring: &mut IoUring,
330    config: &BenchConfig,
331    stats: &BenchStats,
332    done_count: &mut usize,
333) {
334    let conn = &mut conns[key];
335    conn.send_pending = false;
336
337    if result < 0 {
338        let err_code = -result;
339        if err_code != libc::ECONNRESET && err_code != libc::EPIPE {
340            eprintln!(
341                "connection {}: send failed: {}",
342                conn.client_index,
343                std::io::Error::from_raw_os_error(err_code)
344            );
345        }
346        mark_failed(conn, stats, done_count);
347        return;
348    }
349
350    conn.advance_send(result as usize);
351
352    // Continue sending remaining bytes
353    if conn.has_pending_send() {
354        submit_send(ring, key, conn);
355        return;
356    }
357
358    match conn.state {
359        ConnState::Disconnecting => {
360            conn.state = ConnState::Done;
361            stats.clients_done.fetch_add(1, Ordering::Relaxed);
362            *done_count += 1;
363        }
364        ConnState::Publishing => {
365            if conn.try_publish(config) {
366                stats.messages_sent.fetch_add(1, Ordering::Relaxed);
367                submit_send(ring, key, conn);
368            }
369            conn.check_publish_complete();
370        }
371        _ => {}
372    }
373}

375fn handle_recv_complete(
376    conns: &mut Slab<Box<Connection>>,
377    key: usize,
378    result: i32,
379    ring: &mut IoUring,
380    config: &BenchConfig,
381    stats: &BenchStats,
382    done_count: &mut usize,
383) {
384    let conn = &mut conns[key];
385    conn.recv_pending = false;
386
387    if result <= 0 {
388        if result < 0 {
389            let err_code = -result;
390            if err_code != libc::ECONNRESET {
391                eprintln!(
392                    "connection {}: recv error: {}",
393                    conn.client_index,
394                    std::io::Error::from_raw_os_error(err_code)
395                );
396            }
397        }
398        if conn.state != ConnState::Done && conn.state != ConnState::Disconnecting {
399            mark_failed(conn, stats, done_count);
400        }
401        return;
402    }
403
404    let nbytes = result as usize;
405    // SAFETY: recv_buf is owned by the Box<Connection> which has a stable address.
406    // The kernel wrote into recv_buf during the recv operation.
407    // We must clone/copy the data before submitting the next recv on the same buffer.
408    let data: Vec<u8> = conn.recv_buf[..nbytes].to_vec();
409    let events = conn.handle_incoming(&data);
410
411    if !events.is_empty() {
412        let outcome = conn.process_events(events);
413        if outcome.acked > 0 {
414            stats
415                .messages_acked
416                .fetch_add(outcome.acked, Ordering::Relaxed);
417        }
418        if outcome.error {
419            stats.errors.fetch_add(1, Ordering::Relaxed);
420        }
421    }
422
423    let drain_complete = matches!(conn.state, ConnState::Draining) && conn.check_drain_complete();
424
425    match conn.state {
426        ConnState::Publishing => {
427            if conn.has_pending_send() && !conn.send_pending {
428                submit_send(ring, key, conn);
429            } else if !conn.send_pending && conn.try_publish(config) {
430                stats.messages_sent.fetch_add(1, Ordering::Relaxed);
431                submit_send(ring, key, conn);
432            }
433            conn.check_publish_complete();
434            submit_recv(ring, key, conn);
435        }
436        ConnState::MqttConnecting => {
437            if conn.has_pending_send() && !conn.send_pending {
438                submit_send(ring, key, conn);
439            }
440            submit_recv(ring, key, conn);
441        }
442        ConnState::Draining if drain_complete => {
443            if conn.has_pending_send() && !conn.send_pending {
444                submit_send(ring, key, conn);
445            } else if !conn.has_pending_send() {
446                conn.state = ConnState::Done;
447                stats.clients_done.fetch_add(1, Ordering::Relaxed);
448                *done_count += 1;
449            }
450        }
451        ConnState::Draining => {
452            submit_recv(ring, key, conn);
453        }
454        ConnState::Failed | ConnState::Done => {}
455        _ => {
456            submit_recv(ring, key, conn);
457        }
458    }
459}

461fn mark_failed(conn: &mut Connection, stats: &BenchStats, done_count: &mut usize) {
462    if conn.state != ConnState::Failed && conn.state != ConnState::Done {
463        conn.state = ConnState::Failed;
464        stats.errors.fetch_add(1, Ordering::Relaxed);
465        stats.clients_done.fetch_add(1, Ordering::Relaxed);
466        *done_count += 1;
467    }
468}

470fn submit_send(ring: &mut IoUring, key: usize, conn: &mut Connection) {
471    if !conn.has_pending_send() || conn.send_pending {
472        return;
473    }
474    let slice = conn.pending_send_slice();
475    let ptr = slice.as_ptr();
476    let len = slice.len() as u32;
477
478    let send_e = opcode::Send::new(Fd(conn.fd), ptr, len)
479        .build()
480        .user_data(encode_user_data(key, OP_SEND));
481
482    conn.send_pending = true;
483    unsafe {
484        if ring.submission().push(&send_e).is_err() {
485            let _ = ring.submitter().submit();
486            let _ = ring.submission().push(&send_e);
487        }
488    }
489}
490
491fn submit_recv(ring: &mut IoUring, key: usize, conn: &mut Connection) {
492    if conn.recv_pending {
493        return;
494    }
495    let buf_ptr = conn.recv_buf.as_mut_ptr();
496    let buf_len = conn.recv_buf.len() as u32;
497
498    let recv_e = opcode::Recv::new(Fd(conn.fd), buf_ptr, buf_len)
499        .build()
500        .user_data(encode_user_data(key, OP_RECV));
501
502    conn.recv_pending = true;
503    unsafe {
504        if ring.submission().push(&recv_e).is_err() {
505            let _ = ring.submitter().submit();
506            let _ = ring.submission().push(&recv_e);
507        }
508    }
