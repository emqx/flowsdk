26/// Returns a human-readable description for MQTT v5 reason codes
27/// Based on MQTT 5.0 Specification Table 2-6 - Reason Codes
28/// Used across CONNACK, PUBACK, PUBREC, PUBREL, PUBCOMP, SUBACK, UNSUBACK, DISCONNECT, and AUTH packets
29pub fn reason_code_to_string(code: u8) -> &'static str {
30    match code {
31        0x00 => "Success",
32        0x01 => "Granted QoS 1",
33        0x02 => "Granted QoS 2",
34        0x04 => "Disconnect with Will Message",
35        0x10 => "No matching subscribers",
36        0x11 => "No subscription existed",
37        0x18 => "Continue authentication",
38        0x19 => "Re-authenticate",
39        0x80 => "Unspecified error",
40        0x81 => "Malformed Packet",
41        0x82 => "Protocol Error",
42        0x83 => "Implementation specific error",
43        0x84 => "Unsupported Protocol Version",
44        0x85 => "Client Identifier not valid",
45        0x86 => "Bad User Name or Password",
46        0x87 => "Not authorized",
47        0x88 => "Server unavailable",
48        0x89 => "Server busy",
49        0x8A => "Banned",
50        0x8B => "Server shutting down",
51        0x8C => "Bad authentication method",
52        0x8D => "Keep Alive timeout",
53        0x8E => "Session taken over",
54        0x8F => "Topic Filter invalid",
55        0x90 => "Topic Name invalid",
56        0x91 => "Packet Identifier in use",
57        0x92 => "Packet Identifier not found",
58        0x93 => "Receive Maximum exceeded",
59        0x94 => "Topic Alias invalid",
60        0x95 => "Packet too large",
61        0x96 => "Message rate too high",
62        0x97 => "Quota exceeded",
63        0x98 => "Administrative action",
64        0x99 => "Payload format invalid",
65        0x9A => "Retain not supported",
66        0x9B => "QoS not supported",
67        0x9C => "Use another server",
68        0x9D => "Server moved",
69        0x9E => "Shared Subscriptions not supported",
70        0x9F => "Connection rate exceeded",
71        0xA0 => "Maximum connect time",
72        0xA1 => "Subscription Identifiers not supported",
73        0xA2 => "Wildcard Subscriptions not supported",
74        _ => "Unknown reason code",
75    }
76}

287    // Connect to the MQTT broker and wait for CONNACK
288    pub fn connected(&mut self) -> io::Result<ConnectionResult> {
289        // Establish connection to the MQTT broker
290        if let Ok(stream) = TcpStream::connect(self.context.peer.clone()) {
291            // Connection established
292
293            // Initialize ClientSession
294            if self.options.sessionless {
295                self.context.session = None;
296                self.options.clean_start = true;
297            } else if self.context.session.is_none() {
298                self.context.session = Some(ClientSession::new());
299            }
300
301            // Send CONNECT packet with
302            let connect_packet = connectv5::MqttConnect::new(
303                self.options.client_id.clone(),
304                self.options.username.clone(),
305                self.options.password.clone(),
306                self.options.will.clone(),
307                self.options.keep_alive,
308                self.options.clean_start,
309                vec![], // Add properties as needed
310            );
311
312            // stream used, now save it
313            let mqtt_stream = MqttStream::new(stream, 16384, 5);
314            self.context.mqtt_stream = Some(mqtt_stream);
315
316            // Send CONNECT packet
317            self.send_packet(connect_packet)?;
318
319            if let Some(packet) = self.recv_packet()? {
320                match packet {
321                    MqttPacket::ConnAck5(connack) => {
322                        // Always update session_present in context
323                        self.context.session_present = connack.session_present;
324
325                        // If connection successful and session not present, clear session state
326                        if connack.reason_code == 0 && !connack.session_present {
327                            if let Some(sess) = &mut self.context.session {
328                                sess.clear();
329                            }
330                            // Also clear pending operations since session is reset
331                            self.clear_pending_operations();
332                        }
333
334                        // Return the connection result regardless of reason code
335                        return Ok(ConnectionResult {
336                            reason_code: connack.reason_code,
337                            session_present: connack.session_present,
338                            properties: connack.properties,
339                        });
340                    }
341                    _ => {
342                        return Err(io::Error::other("Expected CONNACK packet"));
343                    }
344                }
345            }
346        }
347
348        // If we reach here, no CONNACK was received
349        Err(io::Error::new(
350            io::ErrorKind::UnexpectedEof,
351            "No CONNACK received from broker",
352        ))
353    }

355    // Subscribe to a topic and wait for SUBACK
356    pub fn subscribed(&mut self, topic: &str, qos: u8) -> io::Result<SubscribeResult> {
357        if let Some(session) = &mut self.context.session {
358            let subscription = subscribev5::TopicSubscription {
359                topic_filter: topic.to_string(),
360                qos,
361                no_local: false,
362                retain_as_published: false,
363                retain_handling: 0, //@TODO maybe 2 (NO handling) is more appropriate
364            };
365
366            let packet_id = session.next_packet_id();
367            let subscribe_packet = subscribev5::MqttSubscribe::new(
368                packet_id,
369                vec![subscription],
370                vec![], // Add properties as needed
371            );
372            // Send SUBSCRIBE packet
373            self.send_packet(subscribe_packet)?;
374            // Wait for SUBACK with matching packet ID
375            if let Some(packet) = self.recv_for_packet(
376                |p| matches!(p, MqttPacket::SubAck5(suback) if suback.packet_id == packet_id),
377            )? {
378                match packet {
379                    MqttPacket::SubAck5(suback) => {
380                        return Ok(SubscribeResult {
381                            packet_id: suback.packet_id,
382                            reason_codes: suback.reason_codes,
383                            properties: suback.properties,
384                        });
385                    }
386                    _ => {
387                        // Store unhandled packet for later processing
388                        self.context.unhandled_packets.push(packet);
389                    }
390                }
391            }
392        } else {
393            return Err(io::Error::other("No active session"));
394        }
395
396        Err(io::Error::new(
397            io::ErrorKind::UnexpectedEof,
398            "No SUBACK received from broker",
399        ))
400    }

450    pub fn published(
451        &mut self,
452        topic: &str,
453        payload: &[u8],
454        qos: u8,
455        retain: bool,
456    ) -> io::Result<PublishResult> {
457        if let Some(session) = &mut self.context.session {
458            let packet_id = if qos > 0 {
459                Some(session.next_packet_id())
460            } else {
461                None
462            };
463            let publish_packet = publishv5::MqttPublish::new(
464                qos,
465                topic.to_string(),
466                packet_id,
467                payload.to_vec(),
468                retain,
469                false,
470            );
471            // Send PUBLISH packet
472            self.send_packet(publish_packet)?;
473
474            // Handle PUBACK/PUBREC response for QoS 1/2 if needed
475            match qos {
476                1 => self.receive_for_puback(packet_id),
477                2 => self.handle_qos2(packet_id),
478                _ => {
479                    // QoS 0, no acknowledgment needed
480                    Ok(PublishResult {
481                        packet_id,
482                        reason_code: None,
483                        properties: None,
484                        qos,
485                    })
486                }
487            }
488        } else {
489            Err(io::Error::other("No active session"))
490        }
491    }

493    fn handle_qos2(&mut self, packet_id: Option<u16>) -> Result<PublishResult, io::Error> {
494        let expected_packet_id = packet_id.unwrap();
495        self.recv_for_packet(
496            |p| matches!(p, MqttPacket::PubRec5(pubrec) if pubrec.packet_id == expected_packet_id),
497        )?;
498
499        self.send_packet(pubrelv5::MqttPubRel::new(expected_packet_id, 0, vec![]))?;
500        match self.recv_for_packet(|p| {
501            matches!(p, MqttPacket::PubComp5(pubcomp) if pubcomp.packet_id == expected_packet_id)
502        })? {
503            Some(MqttPacket::PubComp5(pubcomp)) => {
504                // PUBCOMP received
505                Ok(PublishResult {
506                    packet_id: Some(pubcomp.packet_id),
507                    reason_code: Some(pubcomp.reason_code),
508                    properties: Some(pubcomp.properties),
509                    qos: 2,
510                })
511            }
512            _ => {
513                Err(io::Error::new(
514                    io::ErrorKind::UnexpectedEof,
515                    "No PUBCOMP received from broker",
516                ))
517            }
518        }
519    }
520
521    fn receive_for_puback(&mut self, packet_id: Option<u16>) -> Result<PublishResult, io::Error> {
522        let expected_packet_id = packet_id.unwrap();
523        match self.recv_for_packet(
524            |p| matches!(p, MqttPacket::PubAck5(puback) if puback.packet_id == expected_packet_id),
525        )? {
526            Some(MqttPacket::PubAck5(puback)) => {
527                // PUBACK received
528                Ok(PublishResult {
529                    packet_id: Some(puback.packet_id),
530                    reason_code: Some(puback.reason_code), // PUBACK reason code 0 = Success
531                    properties: Some(puback.properties),   // No properties in PUBACK
532                    qos: 1,
533                })
534            }
535            None => Err(io::Error::new(
536                io::ErrorKind::UnexpectedEof,
537                "No PUBACK received from broker",
538            )),
539            _ => unreachable!(),
540        }
541    }

668    /// Send PUBLISH packet without waiting for PUBACK/PUBREC/PUBCOMP
669    pub fn publish_send(
670        &mut self,
671        topic: &str,
672        payload: &[u8],
673        qos: u8,
674        retain: bool,
675    ) -> io::Result<Option<u16>> {
676        if let Some(session) = &mut self.context.session {
677            let packet_id = if qos > 0 {
678                Some(session.next_packet_id())
679            } else {
680                None
681            };
682
683            let publish_packet = publishv5::MqttPublish::new(
684                qos,
685                topic.to_string(),
686                packet_id,
687                payload.to_vec(),
688                retain,
689                false,
690            );
691
692            // Send PUBLISH packet without waiting for acknowledgment
693            self.send_packet(publish_packet)?;
694
695            // For QoS 1/2, track the packet_id in session state for retransmission
696            if let Some(pid) = packet_id {
697                self.context
698                    .pending_publishes
699                    .insert(pid, (topic.to_string(), qos));
700            }
701
702            Ok(packet_id)
703        } else {
704            Err(io::Error::other("No active session"))
705        }
706    }

759    fn send_packet<T>(&mut self, packet: T) -> io::Result<()>
760    where
761        T: MqttControlPacket,
762    {
763        let stream = match &mut self.context.mqtt_stream {
764            Some(s) => s,
765            None => {
766                return Err(io::Error::new(
767                    io::ErrorKind::NotConnected,
768                    "No active MQTT stream connection",
769                ));
770            }
771        };
772
773        let packet_bytes = packet.to_bytes().map_err(|e| {
774            io::Error::new(
775                io::ErrorKind::InvalidData,
776                format!("Failed to serialize packet: {:?}", e),
777            )
778        })?;
779
780        stream.write(&packet_bytes).map_err(|e| {
781            io::Error::new(
782                io::ErrorKind::WriteZero,
783                format!("Failed to write packet: {}", e),
784            )
785        })?;
786
787        Ok(())
788    }

790    pub fn recv_packet(&mut self) -> io::Result<Option<MqttPacket>> {
791        let stream = match &mut self.context.mqtt_stream {
792            Some(s) => s,
793            None => {
794                return Err(io::Error::new(
795                    io::ErrorKind::NotConnected,
796                    "No active MQTT stream connection",
797                ));
798            }
799        };
800
801        match stream.next() {
802            Some(Ok(packet)) => Ok(Some(packet)),
803            Some(Err(parse_error)) => Err(io::Error::new(
804                io::ErrorKind::InvalidData,
805                format!("Failed to parse MQTT packet: {:?}", parse_error),
806            )),
807            None => Ok(None), // End of stream
808        }
809    }

811    pub fn recv_for_packet<F>(&mut self, mut f: F) -> io::Result<Option<MqttPacket>>
812    where
813        F: FnMut(&MqttPacket) -> bool,
814    {
815        let stream = match &mut self.context.mqtt_stream {
816            Some(s) => s,
817            None => {
818                return Err(io::Error::new(
819                    io::ErrorKind::NotConnected,
820                    "No active MQTT stream connection",
821                ));
822            }
823        };
824
825        // find_map handles errors and applies the predicate only to successful packets
826        stream
827            .find_map(|result| match result {
828                Ok(packet) if f(&packet) => Some(Ok(packet)),
829                Ok(other) => {
830                    self.context.unhandled_packets.push(other);
831                    None
832                }
833                Err(parse_err) => Some(Err(io::Error::new(io::ErrorKind::InvalidData, parse_err))),
834            })
835            .transpose() // Convert Option<Result<T, E>> to Result<Option<T>, E>
836    }
