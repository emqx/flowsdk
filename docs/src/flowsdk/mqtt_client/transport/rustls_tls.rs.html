22    /// TLS configuration for Rustls backend
23    #[derive(Clone, Default, Debug)]
24    pub struct RustlsTlsConfig {
25        /// Use system root certificates (platform trust store). Enabled by default.
26        pub use_system_roots: bool,
27        /// Additional custom root certificates (DER bytes)
28        pub custom_root_certs: Vec<Vec<u8>>,
29        /// Optional client certificate chain (DER bytes per cert)
30        pub client_cert_chain: Option<Vec<Vec<u8>>>,
31        /// Optional client private key (DER bytes)
32        pub client_private_key: Option<Vec<u8>>,
33        /// ⚠️ DANGEROUS: Skip certificate chain validation (testing only!)
34        pub danger_accept_invalid_certs: bool,
35        /// ⚠️ DANGEROUS: Skip hostname verification (testing only!)
36        pub danger_accept_invalid_hostnames: bool,
37        /// Optional ALPN protocols (usually not needed for MQTT over TCP)
38        pub alpn_protocols: Vec<Vec<u8>>,
39        /// Enable TLS key logging (writes to the file specified by SSLKEYLOGFILE env var)
40        pub enable_key_log: bool,
41    }

43    impl RustlsTlsConfig {
44        /// Create a new builder for `RustlsTlsConfig`.
45        pub fn builder() -> RustlsTlsConfigBuilder {
46            RustlsTlsConfigBuilder::default()

49        /// Build a rustls ClientConfig from this configuration.
50        pub fn to_client_config(&self) -> Result<RustlsClientConfig, TransportError> {
51            // Prepare root store
52            let mut roots = RootCertStore::empty();
53
54            if self.use_system_roots {
55                let native_certs = rustls_native_certs::load_native_certs().map_err(|e| {
56                    TransportError::Tls(format!("failed to load platform root certs: {}", e))
57                })?;
58                for cert in native_certs {
59                    roots.add(cert).map_err(|e| {
60                        TransportError::Tls(format!("failed to add root cert: {:?}", e))
61                    })?;
62                }
63            }
64
65            // Add any custom roots
66            for cert in &self.custom_root_certs {
67                roots.add(cert.clone().into()).map_err(|e| {
68                    TransportError::Tls(format!("failed to add custom root: {:?}", e))
69                })?;
70            }
71
72            // Build rustls config with or without client auth
73            let mut cfg = if let (Some(chain), Some(key)) = (
74                self.client_cert_chain.clone(),
75                self.client_private_key.clone(),
76            ) {
77                let chain_der: Vec<CertificateDer<'static>> =
78                    chain.into_iter().map(CertificateDer::from).collect();
79                let parsed_key = PrivateKeyDer::try_from(key).map_err(|e| {
80                    TransportError::Tls(format!(
81                        "failed to parse client private key DER for mTLS: {:?}",
82                        e
83                    ))
84                })?;
85                let key_der: PrivateKeyDer<'static> = parsed_key.clone_key();
86
87                RustlsClientConfig::builder()
88                    .with_root_certificates(roots)
89                    .with_client_auth_cert(chain_der, key_der)
90                    .map_err(|e| {
91                        TransportError::Tls(format!(
92                            "failed to build rustls client config with client auth: {}",
93                            e
94                        ))
95                    })?
96            } else {
97                RustlsClientConfig::builder()
98                    .with_root_certificates(roots)
99                    .with_no_client_auth()
100            };
101
102            // Apply ALPN if provided
103            if !self.alpn_protocols.is_empty() {
104                cfg.alpn_protocols = self.alpn_protocols.clone();
105            }
106
107            // Apply dangerous overrides if requested
108            if self.danger_accept_invalid_certs || self.danger_accept_invalid_hostnames {
109                cfg.dangerous()
110                    .set_certificate_verifier(Arc::new(InsecureServerCertVerifier {
111                        skip_name_check: self.danger_accept_invalid_hostnames,
112                    }));
113            }
114
115            // Enable TLS key logging if requested (writes to SSLKEYLOGFILE)
116            if self.enable_key_log {
117                cfg.key_log = Arc::new(rustls::KeyLogFile::new());
118            }
119
120            Ok(cfg)
121        }
122    }

137    impl RustlsTlsConfigBuilder {
138        /// Enable/disable loading platform root certificates (default: true)
139        pub fn use_system_roots(mut self, enable: bool) -> Self {
140            self.use_system_roots = enable;
141            self
142        }

144        /// Add custom root certificates from PEM bytes (may contain multiple CERTIFICATE sections).
145        pub fn add_roots_from_pem(mut self, pem_data: &[u8]) -> Result<Self, TransportError> {
146            let iter = CertificateDer::pem_slice_iter(pem_data);
147            let mut any = false;
148            for c in iter.flatten() {
149                self.custom_root_certs
150                    .push(c.into_owned().as_ref().to_vec());
151                any = true;
152            }
153            if !any {
154                return Err(TransportError::Tls(
155                    "no valid certificates found in PEM data".to_string(),
156                ));
157            }
158            Ok(self)
159        }
160
161        /// Load custom root certificates from a PEM file path.
162        pub fn add_roots_from_pem_file(
163            mut self,
164            file: impl AsRef<std::path::Path>,
165        ) -> Result<Self, TransportError> {
166            match CertificateDer::pem_file_iter(file) {
167                Ok(iter) => {
168                    for item in iter.flatten() {
169                        self.custom_root_certs
170                            .push(item.into_owned().as_ref().to_vec());
171                    }
172                    Ok(self)
173                }
174                Err(e) => Err(TransportError::Tls(format!(
175                    "failed to read/parse root certificates PEM file: {:?}",
176                    e
177                ))),
178            }
179        }
180
181        /// Set client certificate chain and private key from PEM bytes.
182        pub fn client_auth_from_pem(
183            mut self,
184            cert_chain_pem: &[u8],
185            private_key_pem: &[u8],
186        ) -> Result<Self, TransportError> {
187            // Collect all CERTIFICATE sections
188            let chain_iter = CertificateDer::pem_slice_iter(cert_chain_pem);
189            let chain: Vec<Vec<u8>> = chain_iter
190                .filter_map(|r| r.ok().map(|c| c.into_owned().as_ref().to_vec()))
191                .collect();
192            if chain.is_empty() {
193                return Err(TransportError::Tls(
194                    "no valid certificates found in client cert PEM".to_string(),
195                ));
196            }
197            // Parse a single private key
198            use std::io::Cursor;
199            let mut cursor = Cursor::new(private_key_pem);
200            let pk = PrivateKeyDer::from_pem_reader(&mut cursor).map_err(|e| {
201                TransportError::Tls(format!("failed to parse client private key PEM: {:?}", e))
202            })?;
203            let der = pk.clone_key().secret_der().to_vec();
204
205            self.client_cert_chain = Some(chain);
206            self.client_private_key = Some(der);
207            Ok(self)
208        }
209
210        /// Set client certificate chain and private key from PEM files.
211        pub fn client_auth_from_pem_files(
212            mut self,
213            cert_chain_file: impl AsRef<std::path::Path>,
214            private_key_file: impl AsRef<std::path::Path>,
215        ) -> Result<Self, TransportError> {
216            let chain = match CertificateDer::pem_file_iter(cert_chain_file) {
217                Ok(iter) => iter
218                    .filter_map(|r| r.ok().map(|c| c.into_owned().as_ref().to_vec()))
219                    .collect(),
220                Err(e) => {
221                    return Err(TransportError::Tls(format!(
222                        "failed to read/parse client certificate chain: {:?}",
223                        e
224                    )))
225                }
226            };
227            let pk = PrivateKeyDer::from_pem_file(private_key_file).map_err(|e| {
228                TransportError::Tls(format!("failed to parse client private key: {:?}", e))
229            })?;
230            let der = pk.clone_key().secret_der().to_vec();
231
232            self.client_cert_chain = Some(chain);
233            self.client_private_key = Some(der);
234            Ok(self)
235        }

278    /// ⚠️ DANGEROUS: A certificate verifier that accepts all certificates or skips name checks.
279    /// This should ONLY be used for testing and development!
280    #[derive(Debug)]
281    struct InsecureServerCertVerifier {
282        skip_name_check: bool,
283    }
284
285    impl ServerCertVerifier for InsecureServerCertVerifier {
286        fn verify_server_cert(
287            &self,
288            end_entity: &CertificateDer<'_>,
289            _intermediates: &[CertificateDer<'_>],
290            server_name: &ServerName<'_>,
291            _ocsp_response: &[u8],
292            _now: UnixTime,
293        ) -> Result<ServerCertVerified, rustls::Error> {
294            // If skip_name_check is false, we should verify the hostname
295            if !self.skip_name_check {
296                // Perform hostname verification using rustls's built-in verification
297                let cert = rustls::server::ParsedCertificate::try_from(end_entity)?;
298                rustls::client::verify_server_name(&cert, server_name)?;
299            }
300            // Accept any certificate without validation; optionally ignore name
301            Ok(ServerCertVerified::assertion())
302        }
303
304        fn verify_tls12_signature(
305            &self,
306            _message: &[u8],
307            _cert: &CertificateDer<'_>,
308            _dss: &rustls::DigitallySignedStruct,
309        ) -> Result<rustls::client::danger::HandshakeSignatureValid, rustls::Error> {
310            Ok(rustls::client::danger::HandshakeSignatureValid::assertion())
311        }
312
313        fn verify_tls13_signature(
314            &self,
315            _message: &[u8],
316            _cert: &CertificateDer<'_>,
317            _dss: &rustls::DigitallySignedStruct,
318        ) -> Result<rustls::client::danger::HandshakeSignatureValid, rustls::Error> {
319            Ok(rustls::client::danger::HandshakeSignatureValid::assertion())
320        }
321
322        fn supported_verify_schemes(&self) -> Vec<rustls::SignatureScheme> {
323            vec![
324                rustls::SignatureScheme::RSA_PKCS1_SHA256,
325                rustls::SignatureScheme::ECDSA_NISTP256_SHA256,
326                rustls::SignatureScheme::RSA_PKCS1_SHA384,
327                rustls::SignatureScheme::ECDSA_NISTP384_SHA384,
328                rustls::SignatureScheme::RSA_PKCS1_SHA512,
329                rustls::SignatureScheme::ECDSA_NISTP521_SHA512,
330                rustls::SignatureScheme::RSA_PSS_SHA256,
331                rustls::SignatureScheme::RSA_PSS_SHA384,
332                rustls::SignatureScheme::RSA_PSS_SHA512,
333                rustls::SignatureScheme::ED25519,
334            ]
335        }

338    /// Rustls-backed TLS transport for MQTT over TCP
339    pub struct RustlsTlsTransport {
340        stream: RustlsTlsStream<TcpStream>,
341        peer_addr: String,
342    }
343
344    impl RustlsTlsTransport {
345        /// Connect using default configuration (system roots)
346        pub async fn connect_rustls(addr: &str) -> Result<Self, TransportError> {
347            let cfg = RustlsTlsConfig::default();
348            Self::connect_with_config(addr, cfg).await
349        }
350
351        /// Connect using an explicit configuration
352        pub async fn connect_with_config(
353            addr: &str,
354            cfg: RustlsTlsConfig,
355        ) -> Result<Self, TransportError> {
356            // Parse address
357            let (host, socket_addr) = parse_mqtt_address(addr)?;
358
359            // Build rustls client config
360            let client_cfg = cfg.to_client_config()?;
361            let connector = RustlsTlsConnector::from(Arc::new(client_cfg));
362
363            // Resolve server name for SNI
364            let server_name = {
365                use std::convert::TryFrom;
366                match ServerName::try_from(host.clone()) {
367                    Ok(sn) => sn,
368                    Err(_e) => {
369                        return Err(TransportError::InvalidAddress(format!(
370                            "invalid DNS name for TLS: {}",
371                            host
372                        )))
373                    }
374                }
375            };
376
377            // Establish TCP
378            let tcp = TcpStream::connect(&socket_addr).await.map_err(|e| {
379                TransportError::ConnectionFailed(format!("TCP connection failed: {}", e))
380            })?;
381
382            // TLS handshake
383            let tls_stream = connector
384                .connect(server_name, tcp)
385                .await
386                .map_err(|e| TransportError::Tls(format!("TLS handshake failed: {}", e)))?;
387
388            Ok(Self {
389                stream: tls_stream,
390                peer_addr: socket_addr,
391            })
392        }
393
394        /// Access the underlying TLS stream
395        pub fn get_ref(&self) -> &RustlsTlsStream<TcpStream> {
396            &self.stream
397        }
398
399        /// Access the underlying TLS stream mutably
400        pub fn get_mut(&mut self) -> &mut RustlsTlsStream<TcpStream> {
401            &mut self.stream
402        }
403    }

405    #[async_trait]
406    impl Transport for RustlsTlsTransport {
407        async fn connect(addr: &str) -> Result<Self, TransportError>
408        where
409            Self: Sized,
410        {
411            Self::connect_rustls(addr).await
412        }
413
414        async fn close(&mut self) -> Result<(), TransportError> {
415            // Rustls/TCP closes on drop; no explicit shutdown available here
416            Ok(())
417        }
418
419        fn peer_addr(&self) -> Result<String, TransportError> {
420            Ok(self.peer_addr.clone())
421        }
422
423        fn local_addr(&self) -> Result<String, TransportError> {
424            self.stream
425                .get_ref()
426                .0
427                .local_addr()
428                .map(|a| a.to_string())
429                .map_err(TransportError::Io)
430        }
431
432        fn set_nodelay(&self, nodelay: bool) -> Result<(), TransportError> {
433            self.stream
434                .get_ref()
435                .0
436                .set_nodelay(nodelay)
437                .map_err(TransportError::Io)
438        }
439    }
