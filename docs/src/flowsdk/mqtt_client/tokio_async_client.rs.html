36/// Events that can occur during MQTT client operation
37#[derive(Debug)]
38pub enum TokioMqttEvent {
39    /// Connection established with broker
40    Connected(ConnectionResult),
41    /// Disconnected from broker (reason code if available)
42    Disconnected(Option<u8>),
43    /// Message published successfully
44    Published(PublishResult),
45    /// Subscription completed
46    Subscribed(SubscribeResult),
47    /// Unsubscription completed
48    Unsubscribed(UnsubscribeResult),
49    /// Incoming message received from broker
50    MessageReceived(MqttPublish),
51    /// Ping response received
52    PingResponse(PingResult),
53    /// Error occurred during operation (enhanced with MqttClientError)
54    Error(MqttClientError),
55    /// TLS peer certificate received (for certificate validation)
56    PeerCertReceived(Vec<u8>),
57    /// Connection lost (will attempt to reconnect if enabled)
58    ConnectionLost,
59    /// Reconnection attempt started
60    ReconnectAttempt(u32),
61    /// All pending operations cleared (on reconnect with clean_start)
62    PendingOperationsCleared,
63    /// Parse error occurred with details
64    ParseError {
65        /// First 100 bytes of raw data for debugging
66        raw_data: Vec<u8>,
67        /// The parse error that occurred
68        error: ParseError,
69        /// Whether this error is recoverable
70        recoverable: bool,
71    },
72    /// Authentication challenge or response received (MQTT v5 enhanced auth)
73    AuthReceived(AuthResult),
74}

629///     .add_topic("sensors/+/temp", 1)
630///     .with_no_local(true)           // Don't receive own messages

1425/// Async stream for outbound MQTT frame bytes.
1426struct AsyncEgressStream {
1427    sender: mpsc::Sender<Vec<u8>>,
1428    receiver: Option<Receiver<Vec<u8>>>,
1429    capacity: usize,
1430}
1431
1432impl AsyncEgressStream {
1433    fn new(capacity: usize) -> Self {
1434        let (sender, receiver) = mpsc::channel(capacity);
1435        Self {
1436            sender,
1437            receiver: Some(receiver),
1438            capacity,
1439        }
1440    }
1441
1442    /// Non-blocking enqueue for outbound bytes. Returns WouldBlock when channel is full.
1443    #[allow(dead_code)]
1444    fn try_send_bytes(&self, data: Vec<u8>) -> io::Result<()> {
1445        match self.sender.try_send(data) {
1446            Ok(()) => Ok(()),
1447            Err(TrySendError::Full(_)) => Err(io::Error::new(
1448                io::ErrorKind::WouldBlock,
1449                "Egress buffer full",
1450            )),
1451            Err(TrySendError::Closed(_)) => Err(io::Error::new(
1452                io::ErrorKind::BrokenPipe,
1453                "Egress stream closed",
1454            )),
1455        }
1456    }
1457
1458    /// Best-effort enqueue that awaits for capacity when needed.
1459    #[allow(dead_code)]
1460    async fn send_bytes(&self, data: Vec<u8>) -> io::Result<()> {
1461        match self.sender.try_send(data) {
1462            Ok(()) => Ok(()),
1463            Err(TrySendError::Full(data)) => self
1464                .sender
1465                .send(data)
1466                .await
1467                .map_err(|_| io::Error::new(io::ErrorKind::BrokenPipe, "Egress stream closed")),
1468            Err(TrySendError::Closed(_)) => Err(io::Error::new(
1469                io::ErrorKind::BrokenPipe,
1470                "Egress stream closed",
1471            )),
1472        }
1473    }
1474
1475    #[allow(clippy::wrong_self_convention)]
1476    fn take_receiver(&mut self) -> Receiver<Vec<u8>> {
1477        self.receiver.take().expect("egress receiver already taken")
1478    }
1479
1480    fn capacity(&self) -> usize {
1481        self.capacity
1482    }
1483}

1485/// Async stream for inbound MQTT packets, fed by transport reads.
1486struct AsyncIngressStream {
1487    sender: mpsc::Sender<MqttPacket>,
1488    receiver: Option<Receiver<MqttPacket>>,
1489    raw_buffer: BytesMut,
1490    pending_packets: VecDeque<MqttPacket>,
1491    max_pending: usize,
1492}
1493
1494impl AsyncIngressStream {
1495    fn new(max_pending: usize) -> Self {
1496        let (sender, receiver) = mpsc::channel(max_pending);
1497        Self {
1498            sender,
1499            receiver: Some(receiver),
1500            raw_buffer: BytesMut::with_capacity(16 * 1024),
1501            pending_packets: VecDeque::new(),
1502            max_pending,
1503        }
1504    }
1505
1506    fn take_receiver(&mut self) -> Receiver<MqttPacket> {
1507        self.receiver
1508            .take()
1509            .expect("ingress receiver already taken")
1510    }
1511
1512    /// Push raw transport bytes, parse MQTT packets, and forward to consumers.
1513    fn push_raw_data(&mut self, data: &[u8], mqtt_version: u8) -> Result<(), MqttClientError> {
1514        self.raw_buffer.extend_from_slice(data);
1515        self.flush_pending()?;
1516
1517        while let Some((packet, consumed)) = self.try_parse_next_packet(mqtt_version)? {
1518            match self.sender.try_send(packet) {
1519                Ok(()) => {
1520                    self.raw_buffer.advance(consumed);
1521                }
1522                Err(TrySendError::Full(packet)) => {
1523                    if self.pending_packets.len() >= self.max_pending {
1524                        return Err(MqttClientError::BufferFull {
1525                            buffer_type: "ingress buffer".to_string(),
1526                            capacity: self.max_pending,
1527                        });
1528                    }
1529                    self.pending_packets.push_back(packet);
1530                    self.raw_buffer.advance(consumed);
1531                    break;
1532                }
1533                Err(TrySendError::Closed(_)) => {
1534                    return Err(MqttClientError::ChannelClosed {
1535                        channel: "ingress stream".to_string(),
1536                    });
1537                }
1538            }
1539        }
1540        Ok(())
1541    }
1542
1543    /// Allow consumers to wake pending packets after they drain channel capacity.
1544    fn flush_pending(&mut self) -> Result<(), MqttClientError> {
1545        while let Some(packet) = self.pending_packets.pop_front() {
1546            match self.sender.try_send(packet) {
1547                Ok(()) => continue,
1548                Err(TrySendError::Full(packet)) => {
1549                    self.pending_packets.push_front(packet);
1550                    break;
1551                }
1552                Err(TrySendError::Closed(_)) => {
1553                    return Err(MqttClientError::ChannelClosed {
1554                        channel: "ingress stream".to_string(),
1555                    });
1556                }
1557            }
1558        }
1559        Ok(())
1560    }
1561
1562    fn try_parse_next_packet(
1563        &self,
1564        mqtt_version: u8,
1565    ) -> Result<Option<(MqttPacket, usize)>, MqttClientError> {
1566        if self.raw_buffer.is_empty() {
1567            return Ok(None);
1568        }
1569
1570        match MqttPacket::from_bytes_with_version(&self.raw_buffer[..], mqtt_version) {
1571            Ok(ParseOk::Packet(packet, consumed)) => Ok(Some((packet, consumed))),
1572            Ok(ParseOk::Continue(_, _)) => Ok(None),
1573            Ok(ParseOk::TopicName(_, _)) => Err(MqttClientError::ProtocolViolation {
1574                message: "Unexpected ParseOk variant: TopicName".to_string(),
1575            }),
1576            Err(ParseError::More(_, _))
1577            | Err(ParseError::BufferTooShort)
1578            | Err(ParseError::BufferEmpty) => Ok(None), // Need more data
1579            Err(e) => {
1580                // Capture raw data for debugging (first 100 bytes)
1581                let raw_data = self.raw_buffer.iter().take(100).copied().collect();
1582                Err(MqttClientError::PacketParsing {
1583                    parse_error: format!("{:?}", e),
1584                    raw_data,
1585                })
1586            }
1587        }
1588    }
1589}

1796    async fn publish_sync_internal(&self, command: PublishCommand) -> io::Result<PublishResult> {
1797        // QoS 0 messages have no acknowledgment, send fire-and-forget
1798        if command.qos == 0 {
1799            self.send_command(TokioClientCommand::Publish(command))
1800                .await?;

1802                packet_id: None,
1803                reason_code: Some(0),

1807        }

1809        // For QoS 1/2, use oneshot channel to wait for acknowledgment
1810        let (tx, rx) = tokio::sync::oneshot::channel();

1812        self.send_command(TokioClientCommand::PublishSync {
1813            command,
1814            response_tx: tx,
1815        })
1816        .await?;

1818        // Block until response received or channel closed
1819        rx.await.map_err(|_| {
1820            io::Error::new(
1821                io::ErrorKind::BrokenPipe,

2486    /// Main async event loop coordinating command handling and socket I/O.
2487    async fn run(mut self) {
2488        let (mut egress_rx, mut ingress_rx) = self.streams.take_receivers();
2489
2490        loop {
2491            tokio::select! {
2492                // Handle COMMANDS from the client API - highest priority
2493                cmd = self.command_rx.recv() => {
2494                    match cmd {
2495                        Some(command) => {
2496                            if !self.handle_command(command).await {
2497                                break; // Shutdown requested
2498                            }
2499                        }
2500                        None => {
2501                            // Channel closed, shutdown
2502                            break;
2503                        }
2504                    }
2505                }
2506
2507                // EGRESS: Drain channel data and write to the transport
2508                frame = egress_rx.recv() => {
2509                    if let Some(frame) = frame {
2510                        if let Err(mqtt_err) = self.handle_egress_frame(frame, &mut egress_rx).await {
2511                            self.event_handler.on_error(&mqtt_err).await;
2512                            self.handle_connection_lost().await;
2513                            continue;
2514                        }
2515                    } else {
2516                        break;
2517                    }
2518                }
2519
2520                // INGRESS: Read raw bytes from transport and push into ingress stream
2521                read_result = async {
2522                    if let Some(stream) = &mut self.stream {
2523                        let mut buffer = vec![0u8; 4096];
2524                        match stream.read(&mut buffer).await {
2525                            Ok(0) => Ok(None),
2526                            Ok(n) => {
2527                                buffer.truncate(n);
2528                                Ok(Some(buffer))
2529                            }
2530                            Err(e) => Err(e),
2531                        }
2532                    } else {
2533                        std::future::pending::<io::Result<Option<Vec<u8>>>>().await
2534                    }
2535                } => {
2536                    match read_result {
2537                        Ok(Some(bytes)) => {
2538                            if let Err(mqtt_err) = self.streams.push_ingress_data(&bytes, self.mqtt_version) {
2539                                // Emit ParseError event if it's a packet parsing error
2540                                if let MqttClientError::PacketParsing { parse_error, raw_data } = &mqtt_err {
2541                                    let parse_err = ParseError::ParseError(parse_error.clone());
2542                                    self.event_handler.on_parse_error(
2543                                        raw_data,
2544                                        &parse_err,
2545                                        mqtt_err.is_recoverable()
2546                                    ).await;
2547                                }
2548
2549                                self.event_handler.on_error(&mqtt_err).await;
2550                                self.handle_connection_lost().await;
2551                                continue;
2552                            }
2553                        }
2554                        Ok(None) => {

2728    /// Check keep alive and send PINGREQ if needed
2729    ///
2730    /// MQTT Keep-Alive Specification:
2731    /// - The client MUST send a control packet within the Keep Alive period
2732    /// - ANY control packet counts (PUBLISH, SUBSCRIBE, PUBACK, etc.)
2733    /// - PINGREQ is only sent if NO other packet was sent during the period
2734    /// - The broker will disconnect if it doesn't receive ANY packet within 1.5x Keep Alive
2735    async fn check_keep_alive(&mut self) {
2736        if !self.is_connected || self.options.keep_alive == 0 {
2737            return;
2738        }
2739
2740        let keep_alive_duration = Duration::from_secs(self.options.keep_alive as u64);
2741        let time_since_last_send = self.last_packet_sent.elapsed();
2742
2743        // Only send PINGREQ if we haven't sent ANY packet within the keep-alive period
2744        if time_since_last_send >= keep_alive_duration {
2745            // Send PINGREQ to satisfy keep-alive requirement
2746            let pingreq = pingreqv5::MqttPingReq::new();
2747            match pingreq.to_bytes() {
2748                Ok(bytes) => {
2749                    if let Err(e) = self.streams.send_egress(bytes).await {
2750                        let mqtt_err = MqttClientError::from_io_error(e, "send keep-alive PINGREQ");
2751                        self.event_handler.on_error(&mqtt_err).await;
2752                        self.handle_connection_lost().await;
2753                        return; // Don't reset timer if connection lost
2754                    }
2755                    // last_packet_sent will be updated in write_to_transport
2756                }
2757                Err(err) => {
2758                    let mqtt_err = MqttClientError::ProtocolViolation {
2759                        message: format!("Failed to serialize PINGREQ: {:?}", err),
2760                    };
2761                    self.event_handler.on_error(&mqtt_err).await;
2762                    return; // Don't reset timer on error
2763                }
2764            }
2765        }
2766
2767        // Reset timer based on time remaining until next keep-alive check
2768        self.reset_keep_alive_timer();
2769    }

2771    /// Reset the keep-alive timer based on when we last sent a packet
2772    fn reset_keep_alive_timer(&mut self) {
2773        if self.options.keep_alive == 0 {
2774            return;
2775        }
2776
2777        let keep_alive_duration = Duration::from_secs(self.options.keep_alive as u64);
2778        let time_since_last_send = self.last_packet_sent.elapsed();
2779
2780        // Calculate when we need to check again
2781        // If we just sent a packet (including PINGREQ), schedule for full keep-alive duration
2782        let time_until_next_check = if time_since_last_send < keep_alive_duration {
2783            // Schedule for when keep-alive period expires
2784            keep_alive_duration - time_since_last_send
2785        } else {
2786            // Edge case: already past keep-alive (shouldn't happen normally)
2787            // Check immediately by using a very small duration
2788            Duration::from_millis(100)
2789        };
2790
2791        self.keep_alive_timer = Some(Box::pin(tokio::time::sleep(time_until_next_check)));
2792    }

3787    /// Schedule reconnection attempt
3788    async fn schedule_reconnect(&mut self) {
3789        // Early exit if auto-reconnect was disabled during sleep
3790        if !self.config.auto_reconnect {
3791            return;
3792        }
3793
3794        if self.config.max_reconnect_attempts > 0
3795            && self.reconnect_attempts >= self.config.max_reconnect_attempts
3796        {
3797            return; // Max attempts reached
3798        }
3799
3800        self.reconnect_attempts += 1;
3801        self.event_handler
3802            .on_reconnect_attempt(self.reconnect_attempts)
3803            .await;
3804
3805        // Calculate delay with exponential backoff
3806        let delay = std::cmp::min(
3807            self.config.reconnect_delay_ms * (1 << (self.reconnect_attempts - 1)),
3808            self.config.max_reconnect_delay_ms,
3809        );
3810
3811        // Sleep and then attempt reconnect
3812        tokio::time::sleep(Duration::from_millis(delay)).await;
3813        self.handle_connect().await;
3814    }
