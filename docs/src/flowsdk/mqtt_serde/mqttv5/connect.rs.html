30impl MqttConnect {
31    pub fn new(
32        client_id: String,
33        username: Option<String>,
34        password: Option<Vec<u8>>,
35        will: Option<Will>,
36        keep_alive: u16,
37        clean_start: bool,
38        properties: Vec<Property>,
39    ) -> Self {
40        MqttConnect {
41            protocol_name: "MQTT".to_string(),
42            protocol_version: 5,
43            keep_alive,
44            client_id,
45            will,
46            username,
47            password,
48            properties,
49            clean_start,
50        }
51    }
52
53    pub fn is_clean_start(&self) -> bool {
54        self.clean_start
55    }
56
57    pub fn is_keep_alive(&self) -> bool {
58        self.keep_alive > 0
59    }
60
61    fn connect_flags(&self) -> u8 {
62        let mut flags: u8 = 0;
63        if self.username.is_some() {
64            flags |= 0x80;
65        }
66        if self.password.is_some() {
67            flags |= 0x40;
68        }
69        if let Some(will) = &self.will {
70            flags |= 0x04; // Will Flag
71            flags |= will.will_qos << 3;
72            if will.will_retain {
73                flags |= 0x20;
74            }
75        }
76        if self.is_clean_start() {
77            flags |= 0x02;
78        }
79        flags
80    }
81}

83impl MqttControlPacket for MqttConnect {
84    fn control_packet_type(&self) -> u8 {
85        ControlPacketType::CONNECT as u8
86    }
87
88    fn variable_header(&self) -> Result<Vec<u8>, ParseError> {
89        let mut bytes = Vec::new();
90
91        // MQTT 5.0: 3.1.2.1 protocol name
92        bytes.extend(encode_utf8_string(&self.protocol_name)?);
93        // MQTT 5.0: 3.1.2.2 protocol level
94        bytes.push(self.protocol_version);
95
96        // MQTT 5.0: 3.1.2.3 connect flags
97        bytes.push(self.connect_flags());
98        bytes.extend(self.keep_alive.to_be_bytes());
99
100        // MQTT 5.0: 3.1.2.11 properties
101        bytes.extend(encode_properities_hdr(&self.properties)?);
102
103        Ok(bytes)
104    }
105
106    // MQTT 5.0: 3.1.3 Connect Payload
107    fn payload(&self) -> Result<Vec<u8>, parser::ParseError> {
108        let mut bytes = Vec::new();
109
110        // MQTT 5.0: 3.1.3.1 Client Identifier
111        bytes.extend(crate::mqtt_serde::encode_utf8_string(&self.client_id)?);
112
113        // MQTT 5.0: 3.1.3.2 Will Properties
114        if let Some(will) = &self.will {
115            let mut will_properties = vec![];
116
117            // MQTT 5.0: 3.1.3.2.2 - Only add if present
118            if let Some(will_delay_interval) = will.properties.will_delay_interval {
119                will_properties.push(Property::WillDelayInterval(will_delay_interval));
120            }
121
122            // MQTT 5.0: 3.1.3.2.3 - Only add if present
123            if let Some(payload_format_indicator) = will.properties.payload_format_indicator {
124                will_properties.push(Property::PayloadFormatIndicator(payload_format_indicator));
125            }
126
127            if let Some(message_expiry_interval) = &will.properties.message_expiry_interval {
128                // MQTT 5.0: 3.1.3.2.4
129                will_properties.push(Property::MessageExpiryInterval(*message_expiry_interval));
130            }
131            if let Some(content_type) = &will.properties.content_type {
132                // MQTT 5.0: 3.1.3.2.5
133                will_properties.push(Property::ContentType(content_type.clone()));
134            }
135            if let Some(response_topic) = &will.properties.response_topic {
136                // MQTT 5.0: 3.1.3.2.6
137                will_properties.push(Property::ResponseTopic(response_topic.clone()));
138            }
139            if let Some(correlation_data) = &will.properties.correlation_data {
140                // MQTT 5.0: 3.1.3.2.7
141                will_properties.push(Property::CorrelationData(correlation_data.clone()));
142            }
143            // MQTT 5.0: 3.1.3.2.8
144            will_properties.extend(will.properties.user_properties.clone());
145            bytes.extend(encode_properities_hdr(&will_properties)?);
146            bytes.extend(crate::mqtt_serde::encode_utf8_string(&will.will_topic)?);
147            bytes.extend(crate::mqtt_serde::encode_binary_data(&will.will_message)?);
148        }
149
150        // MQTT 5.0: 3.1.3.5 Username
151        if let Some(username) = &self.username {
152            bytes.extend(crate::mqtt_serde::encode_utf8_string(username)?);
153        }
154        // MQTT 5.0: 3.1.3.6 Password
155        if let Some(password) = &self.password {
156            bytes.extend(crate::mqtt_serde::encode_binary_data(password)?);
157        }
158        Ok(bytes)
159    }

161    fn from_bytes(buffer: &[u8]) -> Result<ParseOk, ParseError> {
162        let packet_type = packet_type(buffer)?;
163        if packet_type != ControlPacketType::CONNECT as u8 {
164            return Err(ParseError::InvalidPacketType);
165        }
166
167        let (size, vbi_len) = parse_remaining_length(&buffer[1..])?;
168        let mut offset: usize = 1 + vbi_len;
169
170        let total_len = offset + size;
171
172        if total_len > buffer.len() {
173            let hint = total_len - buffer.len();
174            return Ok(ParseOk::Continue(hint, 0));
175        }
176
177        // MQTT 5.0: 3.1.2 Var hdrs
178        // MQTT 5.0: 3.1.2.1 protocol name
179        let (protocol_name, consumed) =
180            parse_utf8_string(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
181        if protocol_name != "MQTT" {
182            return Err(ParseError::UnSuppProtoVsn);
183        }
184        offset += consumed;
185
186        // MQTT 5.0: 3.1.2.2 protocol version
187        let protocol_version = *buffer.get(offset).ok_or(ParseError::BufferTooShort)?;
188        offset += 1;
189
190        // MQTT 5.0: 3.1.2.3 connect flags, contains
191        // - MQTT 5.0: 3.1.2.4 clean start bit 1
192        // - MQTT 5.0: 3.1.2.5 will flag   bit 2
193        // - MQTT 5.0: 3.1.2.6 will QoS    bit 3-4
194        // - MQTT 5.0: 3.1.2.7 will Retain bit 5
195        // - MQTT 5.0: 3.1.2.8 user name   bit 6
196        // - MQTT 5.0: 3.1.2.9 password    bit 7
197        let connect_flags = *buffer.get(offset).ok_or(ParseError::BufferTooShort)?;
198        offset += 1;
199
200        // [MQTT-3.1.2-3] The Server MUST validate that the reserved flag in the CONNECT packet is set to 0
201        #[cfg(feature = "strict-protocol-compliance")]
202        if connect_flags & 0x01 != 0 {
203            return Err(ParseError::ParseError(
204                "CONNECT packet reserved flag is not 0".to_string(),
205            ));
206        }
207
208        let will_flag = (connect_flags & 0x04) != 0;
209        let will_qos = (connect_flags >> 3) & 0x03;
210        let will_retain = (connect_flags & 0x20) != 0;
211
212        if !will_flag {
213            // [MQTT-3.1.2-11] If the Will Flag is set to 0, then the Will QoS MUST be set to 0
214            #[cfg(feature = "strict-protocol-compliance")]
215            if will_qos != 0 {
216                return Err(ParseError::ParseError(
217                    "Will QoS must be 0 if Will Flag is 0".to_string(),
218                ));
219            }
220            // [MQTT-3.1.2-13] If the Will Flag is set to 0, then Will Retain MUST be set to 0
221            #[cfg(feature = "strict-protocol-compliance")]
222            if will_retain {
223                return Err(ParseError::ParseError(
224                    "Will Retain must be 0 if Will Flag is 0".to_string(),
225                ));
226            }
227        } else {
228            // [MQTT-3.1.2-12] A value of 3 (0x03) is a Malformed Packet.
229            #[cfg(feature = "strict-protocol-compliance")]
230            if will_qos == 3 {
231                return Err(ParseError::ParseError("Will QoS cannot be 3".to_string()));
232            }
233        }
234
235        // MQTT 5.0: 3.1.2.10 keep alive
236        let slice = buffer
237            .get(offset..offset + 2)
238            .ok_or(ParseError::BufferTooShort)?;
239        let keep_alive = u16::from_be_bytes(slice.try_into().unwrap());
240        offset += 2;
241
242        // MQTT 5.0: 3.1.2.11 Properties
243        let (properties, consumed) =
244            parse_properties_hdr(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
245        offset += consumed;
246
247        // MQTT 5.0: 3.1.3 Payload
248        // MQTT 5.0: 3.1.3.1 CLientID
249        let (client_id, consumed) =
250            parse_utf8_string(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
251        offset += consumed;
252
253        // MQTT 5.0: 3.1.3.2 Will Properties
254        // MQTT 5.0: 3.1.3.3 Will Topic
255        // MQTT 5.0: 3.1.3.4 Will Payload
256        let will = if will_flag {
257            // Will is present
258            let (will, consumed) = Will::from_bytes(&buffer[offset..], connect_flags)?;
259            offset += consumed;
260            if will.will_topic.is_empty() || will.will_message.is_empty() {
261                return Err(ParseError::ParseError(
262                    "Will topic or message cannot be empty".to_string(),
263                ));
264            }
265            Some(will)
266        } else {
267            // No Will
268            None
269        };
270
271        // MQTT 5.0: 3.1.3.5 Username
272        let username = if connect_flags & 0x80 != 0 {
273            let (username, consumed) =
274                parse_utf8_string(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
275            offset += consumed;
276            if username.is_empty() {
277                return Err(ParseError::ParseError(
278                    "Username cannot be empty".to_string(),
279                ));
280            }
281            Some(username.to_string())
282        } else {
283            None
284        };
285        // MQTT 5.0: 3.1.3.6 Password
286        let password = if connect_flags & 0x40 != 0 {
287            let (password, consumed) =
288                parse_binary_data(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
289            offset += consumed;
290            if password.is_empty() {
291                return Err(ParseError::ParseError(
292                    "Password cannot be empty".to_string(),
293                ));
294            }
295            Some(password)
296        } else {
297            None
298        };
299
300        let packet = MqttPacket::Connect5(MqttConnect {
301            protocol_name: protocol_name.to_string(),
302            protocol_version,
303            keep_alive,
304            client_id: client_id.to_string(),
305            will,
306            username,
307            password,
308            properties,
309            clean_start: connect_flags & 0x02 != 0,
310        });
311
312        if offset != total_len {
313            return Err(ParseError::InternalError(format!(
314                "Inconsistent offset {} != total: {}",
315                offset, total_len
316            )));
317        }
318
319        Ok(ParseOk::Packet(packet, offset))
320    }
321}
