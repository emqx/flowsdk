59    fn connect_flags(&self) -> u8 {
60        let mut flags: u8 = 0;
61        if self.username.is_some() {
62            flags |= 0x80;
63        }
64        if self.password.is_some() {
65            flags |= 0x40;
66        }
67        if let Some(will) = &self.will {
68            flags |= 0x04; // Will Flag
69            flags |= will.will_qos << 3;
70            if will.will_retain {
71                flags |= 0x20;
72            }
73        }
74        if self.is_clean_start() {
75            flags |= 0x02;
76        }
77        flags
78    }

86    fn variable_header(&self) -> Result<Vec<u8>, ParseError> {
87        let mut bytes = Vec::new();
88
89        // MQTT 5.0: 3.1.2.1 protocol name
90        bytes.extend(encode_utf8_string(&self.protocol_name)?);
91        // MQTT 5.0: 3.1.2.2 protocol level
92        bytes.push(self.protocol_version);
93
94        // MQTT 5.0: 3.1.2.3 connect flags
95        bytes.push(self.connect_flags());
96        bytes.extend(self.keep_alive.to_be_bytes());
97
98        // MQTT 5.0: 3.1.2.11 properties
99        bytes.extend(encode_properities_hdr(&self.properties)?);
100
101        Ok(bytes)
102    }
103
104    // MQTT 5.0: 3.1.3 Connect Payload
105    fn payload(&self) -> Result<Vec<u8>, parser::ParseError> {
106        let mut bytes = Vec::new();
107
108        // MQTT 5.0: 3.1.3.1 Client Identifier
109        bytes.extend(crate::mqtt_serde::encode_utf8_string(&self.client_id)?);
110
111        // MQTT 5.0: 3.1.3.2 Will Properties
112        if let Some(will) = &self.will {
113            let mut will_properties = vec![];
114
115            // MQTT 5.0: 3.1.3.2.2 - Only add if present
116            if let Some(will_delay_interval) = will.properties.will_delay_interval {
117                will_properties.push(Property::WillDelayInterval(will_delay_interval));
118            }
119
120            // MQTT 5.0: 3.1.3.2.3 - Only add if present
121            if let Some(payload_format_indicator) = will.properties.payload_format_indicator {
122                will_properties.push(Property::PayloadFormatIndicator(payload_format_indicator));
123            }
124
125            if let Some(message_expiry_interval) = &will.properties.message_expiry_interval {
126                // MQTT 5.0: 3.1.3.2.4
127                will_properties.push(Property::MessageExpiryInterval(*message_expiry_interval));
128            }
129            if let Some(content_type) = &will.properties.content_type {
130                // MQTT 5.0: 3.1.3.2.5
131                will_properties.push(Property::ContentType(content_type.clone()));
132            }
133            if let Some(response_topic) = &will.properties.response_topic {
134                // MQTT 5.0: 3.1.3.2.6
135                will_properties.push(Property::ResponseTopic(response_topic.clone()));
136            }
137            if let Some(correlation_data) = &will.properties.correlation_data {
138                // MQTT 5.0: 3.1.3.2.7
139                will_properties.push(Property::CorrelationData(correlation_data.clone()));
140            }
141            // MQTT 5.0: 3.1.3.2.8
142            will_properties.extend(will.properties.user_properties.clone());
143            bytes.extend(encode_properities_hdr(&will_properties)?);
144            bytes.extend(crate::mqtt_serde::encode_utf8_string(&will.will_topic)?);
145            bytes.extend(crate::mqtt_serde::encode_binary_data(&will.will_message)?);
146        }
147
148        // MQTT 5.0: 3.1.3.5 Username
149        if let Some(username) = &self.username {
150            bytes.extend(crate::mqtt_serde::encode_utf8_string(username)?);
151        }
152        // MQTT 5.0: 3.1.3.6 Password
153        if let Some(password) = &self.password {
154            bytes.extend(crate::mqtt_serde::encode_binary_data(password)?);
155        }
156        Ok(bytes)
157    }

159    fn from_bytes(buffer: &[u8]) -> Result<ParseOk, ParseError> {
160        let packet_type = packet_type(buffer)?;
161        if packet_type != ControlPacketType::CONNECT as u8 {
162            return Err(ParseError::InvalidPacketType);
163        }
164
165        let (size, vbi_len) = parse_remaining_length(&buffer[1..])?;
166        let mut offset: usize = 1 + vbi_len;
167
168        let total_len = offset + size;
169
170        if total_len > buffer.len() {
171            let hint = total_len - buffer.len();
172            return Ok(ParseOk::Continue(hint, 0));
173        }
174
175        // MQTT 5.0: 3.1.2 Var hdrs
176        // MQTT 5.0: 3.1.2.1 protocol name
177        let (protocol_name, consumed) =
178            parse_utf8_string(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
179        if protocol_name != "MQTT" {
180            return Err(ParseError::UnSuppProtoVsn);
181        }
182        offset += consumed;
183
184        // MQTT 5.0: 3.1.2.2 protocol version
185        let protocol_version = *buffer.get(offset).ok_or(ParseError::BufferTooShort)?;
186        offset += 1;
187
188        // MQTT 5.0: 3.1.2.3 connect flags, contains
189        // - MQTT 5.0: 3.1.2.4 clean start bit 1
190        // - MQTT 5.0: 3.1.2.5 will flag   bit 2
191        // - MQTT 5.0: 3.1.2.6 will QoS    bit 3-4
192        // - MQTT 5.0: 3.1.2.7 will Retain bit 5
193        // - MQTT 5.0: 3.1.2.8 user name   bit 6
194        // - MQTT 5.0: 3.1.2.9 password    bit 7
195        let connect_flags = *buffer.get(offset).ok_or(ParseError::BufferTooShort)?;
196        offset += 1;
197
198        // [MQTT-3.1.2-3] The Server MUST validate that the reserved flag in the CONNECT packet is set to 0
199        #[cfg(feature = "strict-protocol-compliance")]
200        if connect_flags & 0x01 != 0 {
201            return Err(ParseError::ParseError(
202                "CONNECT packet reserved flag is not 0".to_string(),
203            ));
204        }
205
206        let will_flag = (connect_flags & 0x04) != 0;
207        let will_qos = (connect_flags >> 3) & 0x03;
208        let will_retain = (connect_flags & 0x20) != 0;
209
210        if !will_flag {
211            // [MQTT-3.1.2-11] If the Will Flag is set to 0, then the Will QoS MUST be set to 0
212            #[cfg(feature = "strict-protocol-compliance")]
213            if will_qos != 0 {
214                return Err(ParseError::ParseError(
215                    "Will QoS must be 0 if Will Flag is 0".to_string(),
216                ));
217            }
218            // [MQTT-3.1.2-13] If the Will Flag is set to 0, then Will Retain MUST be set to 0
219            #[cfg(feature = "strict-protocol-compliance")]
220            if will_retain {
221                return Err(ParseError::ParseError(
222                    "Will Retain must be 0 if Will Flag is 0".to_string(),
223                ));
224            }
225        } else {
226            // [MQTT-3.1.2-12] A value of 3 (0x03) is a Malformed Packet.
227            #[cfg(feature = "strict-protocol-compliance")]
228            if will_qos == 3 {
229                return Err(ParseError::ParseError("Will QoS cannot be 3".to_string()));
230            }
231        }
232
233        // MQTT 5.0: 3.1.2.10 keep alive
234        let slice = buffer
235            .get(offset..offset + 2)
236            .ok_or(ParseError::BufferTooShort)?;
237        let keep_alive = u16::from_be_bytes(slice.try_into().unwrap());
238        offset += 2;
239
240        // MQTT 5.0: 3.1.2.11 Properties
241        let (properties, consumed) =
242            parse_properties_hdr(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
243        offset += consumed;
244
245        // MQTT 5.0: 3.1.3 Payload
246        // MQTT 5.0: 3.1.3.1 CLientID
247        let (client_id, consumed) =
248            parse_utf8_string(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
249        offset += consumed;
250
251        // MQTT 5.0: 3.1.3.2 Will Properties
252        // MQTT 5.0: 3.1.3.3 Will Topic
253        // MQTT 5.0: 3.1.3.4 Will Payload
254        let will = if will_flag {
255            // Will is present
256            let (will, consumed) = Will::from_bytes(&buffer[offset..], connect_flags)?;
257            offset += consumed;
258            if will.will_topic.is_empty() || will.will_message.is_empty() {
259                return Err(ParseError::ParseError(
260                    "Will topic or message cannot be empty".to_string(),
261                ));
262            }
263            Some(will)
264        } else {
265            // No Will
266            None
267        };
268
269        // MQTT 5.0: 3.1.3.5 Username
270        let username = if connect_flags & 0x80 != 0 {
271            let (username, consumed) =
272                parse_utf8_string(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
273            offset += consumed;
274            if username.is_empty() {
275                return Err(ParseError::ParseError(
276                    "Username cannot be empty".to_string(),
277                ));
278            }
279            Some(username.to_string())
280        } else {
281            None
282        };
283        // MQTT 5.0: 3.1.3.6 Password
284        let password = if connect_flags & 0x40 != 0 {
285            let (password, consumed) =
286                parse_binary_data(buffer.get(offset..).ok_or(ParseError::BufferTooShort)?)?;
287            offset += consumed;
288            if password.is_empty() {
289                return Err(ParseError::ParseError(
290                    "Password cannot be empty".to_string(),
291                ));
292            }
293            Some(password)
294        } else {
295            None
296        };
297
298        let packet = MqttPacket::Connect5(MqttConnect {
299            protocol_name: protocol_name.to_string(),
300            protocol_version,
301            keep_alive,
302            client_id: client_id.to_string(),
303            will,
304            username,
305            password,
306            properties,
307            clean_start: connect_flags & 0x02 != 0,
308        });
309
310        if offset != total_len {
311            return Err(ParseError::InternalError(format!(
312                "Inconsistent offset {} != total: {}",
313                offset, total_len
314            )));
315        }
316
317        Ok(ParseOk::Packet(packet, offset))
318    }
