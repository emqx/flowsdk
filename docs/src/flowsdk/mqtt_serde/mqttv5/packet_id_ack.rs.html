3/// Generates a v5.0 "packet identifier acknowledgement" packet: a struct with
4/// `packet_id: u16`, `reason_code: u8`, and `properties: Vec<Property>` fields,
5/// no payload, and the compact wire format defined in MQTT 5.0 §3.x.2.
6///
7/// Used by PUBACK, PUBREC, PUBCOMP, and PUBREL.
8///
9/// Parameters:
10/// - `$struct_name`:    the generated struct name (e.g. `MqttPubAck`)
11/// - `$packet_type`:    a `ControlPacketType` variant (e.g. `PUBACK`)
12/// - `$flags`:          the fixed-header flags nibble (`0x00` for all except PUBREL which uses `0x02`)
13/// - `$packet_name`:    a string literal for error messages (e.g. `"PUBACK"`)
14/// - `$packet_variant`: the `MqttPacket` enum variant (e.g. `PubAck5`)
15/// - `$strict_flags`:   whether to validate the flags nibble under `strict-protocol-compliance`
16///   (only needed for PUBREL; pass `true` or `false`)
17macro_rules! v5_packet_id_ack {
18    (
19        $struct_name:ident,
20        $packet_type:ident,
21        $flags:expr,
22        $packet_name:literal,
23        $packet_variant:ident,
24        $strict_flags:expr $(,)?
25    ) => {
26        #[derive(Debug, PartialEq, Clone, ::serde::Serialize, ::serde::Deserialize)]
27        #[cfg_attr(feature = "arbitrary", derive(::arbitrary::Arbitrary))]
28        pub struct $struct_name {
29            pub packet_id: u16,
30            pub reason_code: u8,
31            pub properties: Vec<$crate::mqtt_serde::mqttv5::common::properties::Property>,
32        }
33
34        impl $struct_name {
35            pub fn new(
36                packet_id: u16,
37                reason_code: u8,
38                properties: Vec<$crate::mqtt_serde::mqttv5::common::properties::Property>,
39            ) -> Self {
40                Self {
41                    packet_id,
42                    reason_code,
43                    properties,
44                }
45            }
46
47            pub fn new_success(packet_id: u16) -> Self {
48                Self::new(packet_id, 0x00, Vec::new())
49            }
50
51            pub fn new_error(
52                packet_id: u16,
53                reason_code: u8,
54                properties: Vec<$crate::mqtt_serde::mqttv5::common::properties::Property>,
55            ) -> Self {
56                Self::new(packet_id, reason_code, properties)
57            }
58        }
59
60        impl $crate::mqtt_serde::control_packet::MqttControlPacket for $struct_name {
61            fn control_packet_type(&self) -> u8 {
62                $crate::mqtt_serde::control_packet::ControlPacketType::$packet_type as u8
63            }
64
65            fn flags(&self) -> u8 {
66                $flags
67            }
68
69            fn variable_header(&self) -> Result<Vec<u8>, $crate::mqtt_serde::parser::ParseError> {
70                use $crate::mqtt_serde::mqttv5::common::properties::encode_properities_hdr;
71                let mut bytes = self.packet_id.to_be_bytes().to_vec();
72                if self.reason_code == 0x00 && self.properties.is_empty() {
73                    return Ok(bytes);
74                }
75                bytes.push(self.reason_code);
76                bytes.extend(encode_properities_hdr(&self.properties)?);
77                Ok(bytes)
78            }
79
80            fn payload(&self) -> Result<Vec<u8>, $crate::mqtt_serde::parser::ParseError> {
81                Ok(Vec::new())
82            }
83
84            fn from_bytes(
85                buffer: &[u8],
86            ) -> Result<$crate::mqtt_serde::parser::ParseOk, $crate::mqtt_serde::parser::ParseError>
87            {
88                use $crate::mqtt_serde::mqttv5::common::properties::parse_properties_hdr;
89                use $crate::mqtt_serde::parser::{
90                    packet_type, parse_packet_id, parse_remaining_length,
91                };
92
93                let pkt_type = packet_type(buffer)?;
94                if pkt_type
95                    != $crate::mqtt_serde::control_packet::ControlPacketType::$packet_type as u8
96                {
97                    return Err($crate::mqtt_serde::parser::ParseError::InvalidPacketType);
98                }
99
100                #[cfg(feature = "strict-protocol-compliance")]
101                if $strict_flags {
102                    let flags = buffer[0] & 0x0F;
103                    if flags != $flags {
104                        return Err($crate::mqtt_serde::parser::ParseError::ParseError(format!(
105                            "Invalid {} flags: expected 0x{:02x}, got 0x{:02x}",
106                            $packet_name, $flags, flags
107                        )));
108                    }
109                }
110
111                let (size, vbi_len) = parse_remaining_length(&buffer[1..])?;
112                let mut offset: usize = 1 + vbi_len;
113                let total_len = offset + size;
114
115                if total_len > buffer.len() {
116                    return Ok($crate::mqtt_serde::parser::ParseOk::Continue(
117                        total_len - buffer.len(),
118                        0,
119                    ));
120                }
121
122                let (packet_id, consumed) = parse_packet_id(
123                    buffer
124                        .get(offset..)
125                        .ok_or($crate::mqtt_serde::parser::ParseError::BufferTooShort)?,
126                )?;
127                offset += consumed;
128
129                let reason_code = if size > 2 {
130                    let code = *buffer
131                        .get(offset)
132                        .ok_or($crate::mqtt_serde::parser::ParseError::BufferTooShort)?;
133                    offset += 1;
134                    code
135                } else {
136                    0x00
137                };
138
139                let (properties, consumed) = if offset < total_len {
140                    parse_properties_hdr(
141                        buffer
142                            .get(offset..total_len)
143                            .ok_or($crate::mqtt_serde::parser::ParseError::BufferTooShort)?,
144                    )?
145                } else {
146                    (vec![], 0)
147                };
148                offset += consumed;
149
150                if offset != total_len {
151                    return Err($crate::mqtt_serde::parser::ParseError::InternalError(
152                        format!("Inconsistent offset {} != total: {}", offset, total_len),
153                    ));
154                }
155
156                Ok($crate::mqtt_serde::parser::ParseOk::Packet(
157                    $crate::mqtt_serde::control_packet::MqttPacket::$packet_variant($struct_name {
158                        packet_id,
159                        reason_code,
160                        properties,
161                    }),
162                    offset,
163                ))
164            }
165        }
166    };
167}
