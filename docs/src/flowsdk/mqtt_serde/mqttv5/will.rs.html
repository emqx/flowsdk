23#[derive(Debug, PartialEq, Serialize, Deserialize, Clone)]
24#[cfg_attr(feature = "arbitrary", derive(arbitrary::Arbitrary))]
25pub struct Will {
26    pub will_qos: u8,
27    pub will_retain: bool,
28    pub will_topic: String,
29    pub will_message: Vec<u8>,
30    pub properties: WillProperties,
31}
32
33impl Will {
34    pub fn new(will_topic: String, will_message: Vec<u8>, will_qos: u8, will_retain: bool) -> Self {
35        Will {
36            will_qos,
37            will_retain,
38            will_topic,
39            will_message,
40            properties: WillProperties::new(),
41        }
42    }
43
44    pub fn from_bytes(buffer: &[u8], connect_flags: u8) -> Result<(Self, usize), ParseError> {
45        let mut offset = 0;
46
47        // MQTT 5.0: 3.1.3.2 Will Properties
48        let (properties, consumed) = parse_properties_hdr(buffer)?;
49        offset += consumed;
50
51        // MQTT 5.0: 3.1.3.3 Will Topic
52        let (will_topic, consumed) = parse_utf8_string(&buffer[offset..])?;
53        offset += consumed;
54
55        // MQTT 5.0: 3.1.3.4 Will Payload
56        let (will_message, consumed) = parse_binary_data(&buffer[offset..])?;
57        offset += consumed;
58
59        let will_qos = (connect_flags >> 3) & 0x03;
60        let will_retain = (connect_flags & 0x20) != 0;
61
62        let mut will = Will::new(will_topic, will_message, will_qos, will_retain);
63        for property in properties {
64            match property {
65                // MQTT 5.0: 3.1.3.2.2 Will Delay Interval
66                Property::WillDelayInterval(value) => {
67                    will.properties.will_delay_interval = Some(value)
68                }
69                // MQTT 5.0: 3.1.3.2.3 Payload format indicator
70                Property::PayloadFormatIndicator(value) => {
71                    will.properties.payload_format_indicator = Some(value)
72                }
73                // MQTT 5.0: 3.1.3.2.4 Message Expiry Interval
74                Property::MessageExpiryInterval(value) => {
75                    will.properties.message_expiry_interval = Some(value);
76                }
77                // MQTT 5.0: 3.1.3.2.5 Content Type
78                Property::ContentType(value) => will.properties.content_type = Some(value),
79                // MQTT 5.0: 3.1.3.2.6 Response Topic
80                Property::ResponseTopic(value) => will.properties.response_topic = Some(value),
81                // MQTT 5.0: 3.1.3.2.7 Correlation Data
82                Property::CorrelationData(value) => will.properties.correlation_data = Some(value),
83                // MQTT 5.0: 3.1.3.2.8 UserProperty
84                _ => {
85                    if let Property::UserProperty(key, value) = property {
86                        will.properties
87                            .user_properties
88                            .push(Property::UserProperty(key, value));
89                    }
90                }
91            }
92        }
93
94        Ok((will, offset))
95    }
