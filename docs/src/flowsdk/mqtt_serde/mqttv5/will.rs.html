21impl Will {
22    pub fn new(will_topic: String, will_message: Vec<u8>, will_qos: u8, will_retain: bool) -> Self {
23        Will {

38    pub fn from_bytes(buffer: &[u8], connect_flags: u8) -> Result<(Self, usize), ParseError> {
39        let mut offset = 0;
40
41        // MQTT 5.0: 3.1.3.2 Will Properties
42        let (properties, consumed) = parse_properties_hdr(buffer)?;
43        offset += consumed;
44
45        // MQTT 5.0: 3.1.3.3 Will Topic
46        let (will_topic, consumed) = parse_utf8_string(&buffer[offset..])?;
47        offset += consumed;
48
49        // MQTT 5.0: 3.1.3.4 Will Payload
50        let (will_message, consumed) = parse_binary_data(&buffer[offset..])?;
51        offset += consumed;
52
53        let will_qos = (connect_flags >> 3) & 0x03;
54        let will_retain = (connect_flags & 0x20) != 0;
55
56        let mut will = Will::new(will_topic, will_message, will_qos, will_retain);
57        for property in properties {
58            match property {
59                // MQTT 5.0: 3.1.3.2.2 Will Delay Interval
60                Property::WillDelayInterval(value) => will.will_delay_interval = value,
61                // MQTT 5.0: 3.1.3.2.3 Payload format indicator
62                Property::PayloadFormatIndicator(value) => will.payload_format_indicator = value,
63                // MQTT 5.0: 3.1.3.2.4 Message Expiry Interval
64                Property::MessageExpiryInterval(value) => will.message_expiry_interval = value,
65                // MQTT 5.0: 3.1.3.2.5 Content Type
66                Property::ContentType(value) => will.content_type = Some(value),
67                // MQTT 5.0: 3.1.3.2.6 Response Topic
68                Property::ResponseTopic(value) => will.response_topic = Some(value),
69                // MQTT 5.0: 3.1.3.2.7 Correlation Data
70                Property::CorrelationData(value) => will.correlation_data = Some(value),
71                // MQTT 5.0: 3.1.3.2.8 UserProperty
72                _ => {
73                    if let Property::UserProperty(key, value) = property {
74                        will.user_properties
75                            .push(Property::UserProperty(key, value));
76                    }
77                }
78            }
79        }
80
81        Ok((will, offset))
82    }
