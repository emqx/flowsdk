8/// Represents the PINGRESP packet in MQTT v5.0.
9///
10/// The PINGRESP packet is sent by the Server to the Client in response to a PINGREQ packet.
11/// It indicates that the Server is alive.
12#[derive(Debug, PartialEq, Clone, Serialize, Deserialize)]
13#[cfg_attr(feature = "arbitrary", derive(arbitrary::Arbitrary))]
14pub struct MqttPingResp;
15
16impl MqttPingResp {
17    /// Creates a new PINGRESP packet.
18    pub fn new() -> Self {
19        Self
20    }
21}
22
23impl Default for MqttPingResp {
24    fn default() -> Self {
25        Self::new()
26    }
27}
28
29impl MqttControlPacket for MqttPingResp {
30    fn control_packet_type(&self) -> u8 {
31        ControlPacketType::PINGRESP as u8
32    }
33
34    fn variable_header(&self) -> Result<Vec<u8>, ParseError> {
35        // MQTT 5.0: 3.13.2 PINGRESP Variable Header
36        // The PINGRESP packet has no Variable Header.
37        Ok(Vec::new())
38    }
39
40    fn payload(&self) -> Result<Vec<u8>, ParseError> {
41        // MQTT 5.0: 3.13.3 PINGRESP Payload
42        // The PINGRESP packet has no Payload.
43        Ok(Vec::new())
44    }
45
46    fn from_bytes(buffer: &[u8]) -> Result<ParseOk, ParseError> {
47        let packet_type = packet_type(buffer)?;
48        if packet_type != ControlPacketType::PINGRESP as u8 {
49            return Err(ParseError::InvalidPacketType);
50        }
51
52        let (size, vbi_len) = parse_remaining_length(&buffer[1..])?;
53        let offset = 1 + vbi_len;
54        let total_len = offset + size;
55
56        if total_len > buffer.len() {
57            return Ok(ParseOk::Continue(total_len - buffer.len(), 0));
58        }
59
60        // MQTT 5.0: 3.13.1 PINGRESP Fixed Header
61        // The Remaining Length field for the PINGRESP packet MUST be zero.
62        #[cfg(feature = "strict-protocol-compliance")]
63        if size != 0 {
64            return Err(ParseError::ParseError(
65                "PINGRESP packet must have remaining length of 0".to_string(),
66            ));
67        }
68
69        // MQTT 5.0: 3.13.1 Reserved bits in the Fixed Header
70        // Bits 3,2,1 and 0 of the Fixed Header in the PINGRESP packet are reserved and MUST be set to 0,0,0,0.
71        #[cfg(feature = "strict-protocol-compliance")]
72        {
73            let flags = buffer[0] & 0x0F;
74            if flags != 0x00 {
75                return Err(ParseError::ParseError(
76                    "PINGRESP packet has invalid fixed header flags".to_string(),
77                ));
78            }
79        }
80
81        let ping_resp = MqttPingResp::new();
82        Ok(ParseOk::Packet(MqttPacket::PingResp5(ping_resp), total_len))
83    }
84}

91    #[test]

109    #[test]
110    fn test_pingresp_serialization() {
111        let pingresp = MqttPingResp::new();
112        let bytes = pingresp.to_bytes().unwrap();
113
114        // Expected: Fixed header only: 0xD0 (packet type 13, flags 0) + 0x00 (remaining length 0)
115        assert_eq!(bytes, vec![0xD0, 0x00]);
116    }
117
118    #[test]
119    fn test_pingresp_deserialization() {
120        let bytes = vec![0xD0, 0x00]; // PINGRESP with remaining length 0
121
122        match MqttPingResp::from_bytes(&bytes).unwrap() {
123            ParseOk::Packet(MqttPacket::PingResp5(_), consumed) => {
124                assert_eq!(consumed, 2);
125            }
126            _ => panic!("Expected PINGRESP packet"),
127        }
128    }
129
130    #[test]
131    fn test_pingresp_roundtrip() {
132        let original = MqttPingResp::new();
133        let bytes = original.to_bytes().unwrap();
134
135        match MqttPingResp::from_bytes(&bytes).unwrap() {
136            ParseOk::Packet(MqttPacket::PingResp5(parsed), _) => {
137                assert_eq!(original, parsed);
138            }
139            _ => panic!("Expected PINGRESP packet"),
140        }
141    }

172            Ok(ParseOk::Continue(needed, _)) => {

176                // This is also acceptable for incomplete packets

205    #[test]
206    fn test_ping_request_response_workflow() {
207        // Test that PINGREQ and PINGRESP work together
208        let pingreq = MqttPingReq::new();
209        let pingreq_bytes = pingreq.to_bytes().unwrap();
210
211        // Simulate server receiving PINGREQ and responding with PINGRESP
212        let pingresp = MqttPingResp::new();
213        let pingresp_bytes = pingresp.to_bytes().unwrap();
214
215        // Verify both packets serialize correctly
216        assert_eq!(pingreq_bytes, vec![0xC0, 0x00]);
217        assert_eq!(pingresp_bytes, vec![0xD0, 0x00]);
218
219        // Verify both can be parsed back
220        match MqttPingReq::from_bytes(&pingreq_bytes).unwrap() {
221            ParseOk::Packet(MqttPacket::PingReq5(_), _) => {}
222            _ => panic!("Expected PINGREQ"),
223        }
224
225        match MqttPingResp::from_bytes(&pingresp_bytes).unwrap() {
226            ParseOk::Packet(MqttPacket::PingResp5(_), _) => {}
227            _ => panic!("Expected PINGRESP"),
228        }
229    }
