186pub trait MqttControlPacket {
187    // MQTT 5.0: 2.1.2, MQTT control packet type
188    fn control_packet_type(&self) -> u8;
189
190    // MQTT 5.0: 2.1.3, Flags in the fixed header
191    fn flags(&self) -> u8 {
192        0u8
193    }
194
195    // Default implementations
196    // Constructs the fixed header for the MQTT packet.
197    // The fixed header consists of a control packet type, flags, and the remaining length.
198    fn fixed_header(&self, len: usize) -> Vec<u8> {
199        let byte1: u8 = (self.control_packet_type()) << 4 | self.flags();
200        let variable_length = encode_variable_length(len);
201        let mut hdr = vec![byte1];
202        hdr.extend(variable_length);
203        hdr
204    }
205
206    // return variable header
207    fn variable_header(&self) -> Result<Vec<u8>, ParseError>;
208
209    // return payload
210    fn payload(&self) -> Result<Vec<u8>, ParseError>;
211
212    // decoder
213    fn from_bytes(buffer: &[u8]) -> Result<ParseOk, ParseError>;
214
215    // encoder
216    fn to_bytes(&self) -> Result<Vec<u8>, ParseError> {
217        let mut bytes = Vec::new();
218
219        let vhdr = self.variable_header()?;
220        let payload = self.payload()?;
221        let remaining_length = vhdr.len() + payload.len();
222        bytes.extend(self.fixed_header(remaining_length));
223        bytes.extend(vhdr);
224        bytes.extend(payload);
225        Ok(bytes)
226    }
227}
