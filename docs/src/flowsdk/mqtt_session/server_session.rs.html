3use crate::mqtt_serde::control_packet::MqttPacket;
4use crate::mqtt_serde::mqttv5::puback::MqttPubAck;
5use crate::mqtt_serde::mqttv5::pubcomp::MqttPubComp;
6use crate::mqtt_serde::mqttv5::publish::MqttPublish;
7use crate::mqtt_serde::mqttv5::pubrec::MqttPubRec;
8use crate::mqtt_serde::mqttv5::pubrel::MqttPubRel;
9use crate::mqtt_serde::mqttv5::subscribe::{MqttSubscribe, TopicSubscription};
10use crate::mqtt_serde::mqttv5::unsubscribe::MqttUnsubscribe;
11use std::collections::HashMap;
12
13pub struct ServerSession {
14    // The client's subscriptions. The key is the topic filter.
15    subscriptions: HashMap<String, TopicSubscription>,
16
17    // QoS 1 and QoS 2 messages that have been sent to the client but not acknowledged.
18    // The key is the packet identifier.
19    unacknowledged_publishes: HashMap<u16, MqttPublish>,
20
21    // QoS 1 and QoS 2 messages pending transmission to the client.
22    pending_publishes: Vec<MqttPublish>,
23
24    // QoS 2 PUBREL messages that have been sent but not yet acknowledged with PUBCOMP.
25    // The key is the packet identifier.
26    unacknowledged_pubrels: HashMap<u16, MqttPubRel>,
27
28    // The Will Message.
29    #[allow(dead_code)]
30    will: Option<MqttPublish>,
31
32    // The session expiry interval.
33    #[allow(dead_code)]
34    session_expiry_interval: u32,
35
36    // The client's receive maximum value.
37    receive_maximum: u16,
38
39    // The number of messages sent in the current window.
40    inflight_messages: u16,
41}
42
43impl ServerSession {
44    pub fn new(receive_maximum: u16) -> Self {
45        ServerSession {
46            subscriptions: HashMap::new(),
47            unacknowledged_publishes: HashMap::new(),
48            pending_publishes: Vec::new(),
49            unacknowledged_pubrels: HashMap::new(),
50            will: None,
51            session_expiry_interval: 0,
52            receive_maximum,
53            inflight_messages: 0,
54        }
55    }
56
57    pub fn handle_incoming_subscribe(&mut self, subscribe: MqttSubscribe) {
58        for subscription in subscribe.subscriptions {
59            self.subscriptions
60                .insert(subscription.topic_filter.clone(), subscription);
61        }
62    }
63
64    pub fn handle_incoming_unsubscribe(&mut self, unsubscribe: MqttUnsubscribe) {
65        for topic_filter in unsubscribe.topic_filters {
66            self.subscriptions.remove(&topic_filter);
67        }
68    }
69
70    pub fn handle_incoming_publish(&mut self, publish: MqttPublish) -> Option<MqttPacket> {
71        // TODO: Implement topic matching logic here.
72        // For now, we will just queue the message for all subscribers.
73        for _subscription in self.subscriptions.values() {
74            self.pending_publishes.push(publish.clone());
75        }
76
77        match publish.qos {
78            1 => Some(MqttPacket::PubAck5(
79                crate::mqtt_serde::mqttv5::puback::MqttPubAck {
80                    packet_id: publish.packet_id.unwrap(),
81                    reason_code: 0,
82                    properties: Vec::new(),
83                },
84            )),
85            2 => {
86                let pubrec = MqttPubRec {
87                    packet_id: publish.packet_id.unwrap(),
88                    reason_code: 0,
89                    properties: Vec::new(),
90                };
91                Some(MqttPacket::PubRec5(pubrec))
92            }
93            _ => None,
94        }
95    }
96
97    pub fn handle_incoming_puback(&mut self, _puback: MqttPubAck) {
98        if self.inflight_messages > 0 {
99            self.inflight_messages -= 1;
100        }
101        self.unacknowledged_publishes.remove(&_puback.packet_id);
102    }
103
104    pub fn handle_incoming_pubrec(&mut self, pubrec: MqttPubRec) -> Option<MqttPubRel> {
105        if pubrec.reason_code < 0x80 {
106            let pubrel = MqttPubRel {
107                packet_id: pubrec.packet_id,
108                reason_code: 0,
109                properties: Vec::new(),
110            };
111            self.unacknowledged_pubrels
112                .insert(pubrec.packet_id, pubrel.clone());
113            self.unacknowledged_publishes.remove(&pubrec.packet_id);
114            Some(pubrel)
115        } else {
116            self.unacknowledged_publishes.remove(&pubrec.packet_id);
117            None
118        }
119    }
120
121    pub fn handle_incoming_pubrel(&mut self, pubrel: MqttPubRel) -> MqttPubComp {
122        MqttPubComp {
123            packet_id: pubrel.packet_id,
124            reason_code: 0,
125            properties: Vec::new(),
126        }
127    }
128
129    pub fn handle_incoming_pubcomp(&mut self, pubcomp: MqttPubComp) {
130        self.unacknowledged_pubrels.remove(&pubcomp.packet_id);
131        if self.inflight_messages > 0 {
132            self.inflight_messages -= 1;
133        }
134    }
135
136    pub fn resend_pending_messages(&mut self) -> Vec<MqttPacket> {
137        let mut packets_to_resend = Vec::new();
138        let mut available_slots = self.receive_maximum.saturating_sub(self.inflight_messages);
139
140        // Resend unacknowledged publishes
141        for (packet_id, publish) in &self.unacknowledged_publishes {
142            if !self.unacknowledged_pubrels.contains_key(packet_id) {
143                if available_slots == 0 {
144                    break;
145                }
146                packets_to_resend.push(MqttPacket::Publish5(publish.clone()));
147                available_slots -= 1;
148            }
149        }
150
151        // Resend unacknowledged pubrels
152        for pubrel in self.unacknowledged_pubrels.values() {
153            if available_slots == 0 {
154                break;
155            }
156            packets_to_resend.push(MqttPacket::PubRel5(pubrel.clone()));
157            available_slots -= 1;
158        }
159
160        // Send pending publishes
161        let mut i = 0;
162        while i < self.pending_publishes.len() && available_slots > 0 {
163            let publish = self.pending_publishes.remove(i);
164            packets_to_resend.push(MqttPacket::Publish5(publish));
165            available_slots -= 1;
166            i += 1;
167        }
168
169        self.inflight_messages = self.receive_maximum - available_slots;
170
171        packets_to_resend
172    }
173}
