11pub struct ServerSession {
12    // The client's subscriptions. The key is the topic filter.
13    subscriptions: HashMap<String, TopicSubscription>,
14
15    // QoS 1 and QoS 2 messages that have been sent to the client but not acknowledged.
16    // The key is the packet identifier.
17    unacknowledged_publishes: HashMap<u16, MqttPublish>,
18
19    // QoS 1 and QoS 2 messages pending transmission to the client.
20    pending_publishes: Vec<MqttPublish>,
21
22    // QoS 2 PUBREL messages that have been sent but not yet acknowledged with PUBCOMP.
23    // The key is the packet identifier.
24    unacknowledged_pubrels: HashMap<u16, MqttPubRel>,
25
26    // The Will Message.
27    #[allow(dead_code)]
28    will: Option<MqttPublish>,
29
30    // The session expiry interval.
31    #[allow(dead_code)]
32    session_expiry_interval: u32,
33
34    // The client's receive maximum value.
35    receive_maximum: u16,
36
37    // The number of messages sent in the current window.
38    inflight_messages: u16,
39}
40
41impl ServerSession {
42    pub fn new(receive_maximum: u16) -> Self {
43        ServerSession {
44            subscriptions: HashMap::new(),
45            unacknowledged_publishes: HashMap::new(),
46            pending_publishes: Vec::new(),
47            unacknowledged_pubrels: HashMap::new(),
48            will: None,
49            session_expiry_interval: 0,
50            receive_maximum,
51            inflight_messages: 0,
52        }
53    }
54
55    pub fn handle_incoming_subscribe(&mut self, subscribe: MqttSubscribe) {
56        for subscription in subscribe.subscriptions {
57            self.subscriptions
58                .insert(subscription.topic_filter.clone(), subscription);
59        }
60    }
61
62    pub fn handle_incoming_unsubscribe(&mut self, unsubscribe: MqttUnsubscribe) {
63        for topic_filter in unsubscribe.topic_filters {
64            self.subscriptions.remove(&topic_filter);
65        }
66    }
67
68    pub fn handle_incoming_publish(&mut self, publish: MqttPublish) -> Option<MqttPacket> {
69        // TODO: Implement topic matching logic here.
70        // For now, we will just queue the message for all subscribers.
71        for _subscription in self.subscriptions.values() {
72            self.pending_publishes.push(publish.clone());
73        }
74
75        match publish.qos {
76            1 => Some(MqttPacket::PubAck5(
77                crate::mqtt_serde::mqttv5::puback::MqttPubAck {
78                    packet_id: publish.packet_id.unwrap(),
79                    reason_code: 0,
80                    properties: Vec::new(),
81                },
82            )),
83            2 => {
84                let pubrec = MqttPubRec {
85                    packet_id: publish.packet_id.unwrap(),
86                    reason_code: 0,
87                    properties: Vec::new(),
88                };
89                Some(MqttPacket::PubRec5(pubrec))
90            }
91            _ => None,
92        }
93    }
94
95    pub fn handle_incoming_puback(&mut self, _puback: MqttPubAck) {
96        if self.inflight_messages > 0 {
97            self.inflight_messages -= 1;
98        }
99        self.unacknowledged_publishes.remove(&_puback.packet_id);
100    }
101
102    pub fn handle_incoming_pubrec(&mut self, pubrec: MqttPubRec) -> Option<MqttPubRel> {
103        if pubrec.reason_code < 0x80 {
104            let pubrel = MqttPubRel {
105                packet_id: pubrec.packet_id,
106                reason_code: 0,
107                properties: Vec::new(),
108            };
109            self.unacknowledged_pubrels
110                .insert(pubrec.packet_id, pubrel.clone());
111            self.unacknowledged_publishes.remove(&pubrec.packet_id);
112            Some(pubrel)
113        } else {
114            self.unacknowledged_publishes.remove(&pubrec.packet_id);
115            None
116        }
117    }
118
119    pub fn handle_incoming_pubrel(&mut self, pubrel: MqttPubRel) -> MqttPubComp {
120        MqttPubComp {
121            packet_id: pubrel.packet_id,
122            reason_code: 0,
123            properties: Vec::new(),
124        }
125    }
126
127    pub fn handle_incoming_pubcomp(&mut self, pubcomp: MqttPubComp) {
128        self.unacknowledged_pubrels.remove(&pubcomp.packet_id);
129        if self.inflight_messages > 0 {
130            self.inflight_messages -= 1;
131        }
132    }
133
134    pub fn resend_pending_messages(&mut self) -> Vec<MqttPacket> {
135        let mut packets_to_resend = Vec::new();
136        let mut available_slots = self.receive_maximum.saturating_sub(self.inflight_messages);
137
138        // Resend unacknowledged publishes
139        for (packet_id, publish) in &self.unacknowledged_publishes {
140            if !self.unacknowledged_pubrels.contains_key(packet_id) {
141                if available_slots == 0 {
142                    break;
143                }
144                packets_to_resend.push(MqttPacket::Publish5(publish.clone()));
145                available_slots -= 1;
146            }
147        }
148
149        // Resend unacknowledged pubrels
150        for pubrel in self.unacknowledged_pubrels.values() {
151            if available_slots == 0 {
152                break;
153            }
154            packets_to_resend.push(MqttPacket::PubRel5(pubrel.clone()));
155            available_slots -= 1;
156        }
157
158        // Send pending publishes
159        let mut i = 0;
160        while i < self.pending_publishes.len() && available_slots > 0 {
161            let publish = self.pending_publishes.remove(i);
162            packets_to_resend.push(MqttPacket::Publish5(publish));
163            available_slots -= 1;
164            i += 1;
165        }
166
167        self.inflight_messages = self.receive_maximum - available_slots;
168
169        packets_to_resend
170    }
171}
