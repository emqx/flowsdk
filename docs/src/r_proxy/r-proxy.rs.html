49async fn run_proxy(
50    listen_port: u16,
51    destination: SocketAddr,
52) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
53    let listener = TcpListener::bind(("0.0.0.0", listen_port)).await?;
54    info!("Listening on 0.0.0.0:{}", listen_port);
55
56    // Create shared gRPC channel (Phase 1: Channel Reuse)
57    // This channel will be reused by all MQTT client connections
58    info!(
59        "Establishing shared gRPC connection to s-proxy at {}",
60        destination
61    );
62    let channel = tonic::transport::Channel::from_shared(format!("http://{}", destination))?
63        .keep_alive_while_idle(true)
64        .keep_alive_timeout(std::time::Duration::from_secs(30))
65        .connect_lazy(); // Use connect_lazy for efficient connection pooling
66
67    let grpc_client = Arc::new(MqttRelayServiceClient::new(channel));
68    info!("Shared gRPC channel established and ready for client connections");
69
70    let proxy_state = Arc::new(RProxyState {
71        connections: Arc::new(DashMap::new()),
72        sequence_counter: AtomicU64::new(1),
73        grpc_client, // Share this client across all MQTT connections
74    });
75
76    loop {
77        info!("Accepting connection...");
78        let (incoming_stream, addr) = listener.accept().await?;
79        let state = proxy_state.clone();
80        info!("Accepted connection from {}", addr);
81        spawn(async move {
82            match handle_new_incoming_tcp(incoming_stream, state).await {
83                Ok(_) => {
84                    info!("Client connection handled successfully");
85                }
86                Err(e) => {
87                    error!("Error handling client connection: {}", e);
88                }
89            }
90        });
91    }
92}

94/// Setup a new gRPC bidirectional stream using the shared channel
95///
96/// Phase 1: This function reuses the shared gRPC channel instead of creating
97/// a new connection for each MQTT client. Each client gets its own isolated
98/// bidirectional stream over the shared channel.
99async fn setup_grpc_streaming(
100    grpc_client: Arc<MqttRelayServiceClient<tonic::transport::Channel>>,
101    session_id: String,
102    client_id: String,
103) -> Result<
104    (
105        mpsc::Sender<MqttStreamMessage>,
106        Streaming<MqttStreamMessage>,
107    ),
108    Box<dyn std::error::Error + Send + Sync>,
109> {
110    // Clone the shared client (cheap Arc clone, shares underlying channel)
111    let mut client = (*grpc_client).clone();
112
113    // Create bidirectional streaming channels for THIS MQTT client
114    let (grpc_stream_sender, grpc_stream_receiver) = mpsc::channel(1000);
115
116    // Establish NEW streaming RPC (but reuses underlying channel)
117    let outbound_gstream = ReceiverStream::new(grpc_stream_receiver);
118    let mut request = Request::new(outbound_gstream);
119
120    // Add session_id and client_id as gRPC metadata (HTTP/2 headers)
121    let metadata = request.metadata_mut();
122    metadata.insert("session-id", session_id.parse().unwrap());
123    metadata.insert("client-id", client_id.parse().unwrap());
124
125    let inbound_gstream = client.stream_mqtt_messages(request).await?.into_inner();
126
127    Ok((grpc_stream_sender, inbound_gstream))
128}

130async fn handle_new_incoming_tcp(
131    incoming: TcpStream,
132    state: Arc<RProxyState>,
133) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
134    // Split stream to read and write halves
135    let (mut in_read_half, out_write_half) = incoming.into_split();
136
137    // Wait for MQTT Connect packet
138    // @TODO: with some timeout
139    let mut parser: MqttParser = MqttParser::new(16384, 0);
140    let connect = wait_for_mqtt_connect(&mut in_read_half, &mut parser).await?;
141
142    let client_id: String = connect.client_id().to_string();
143    let mqtt_version = connect.version();
144    let session_id: String = format!(
145        "{}-{}",
146        client_id,
147        state.sequence_counter.fetch_add(1, Ordering::SeqCst)
148    );
149
150    info!(
151        "New MQTT client connecting: {} (session: {})",
152        client_id, session_id
153    );
154
155    // Setup gRPC streaming using shared channel (Phase 1: Channel Reuse)
156    // Session metadata will be sent via gRPC metadata (HTTP/2 headers)
157    debug!(
158        "Setting up gRPC stream for client {} using shared channel",
159        client_id
160    );
161    let (gstream_sender, inbound_gstream) = setup_grpc_streaming(
162        state.grpc_client.clone(),
163        session_id.clone(),
164        client_id.clone(),
165    )
166    .await?;
167    debug!(
168        "gRPC stream established for client {} (reused shared channel) with session metadata",
169        client_id
170    );
171
172    let control_type = match mqtt_version {
173        5 => mqtt_unified_pb::session_control::ControlType::Establishv5,
174        3 => mqtt_unified_pb::session_control::ControlType::Establishv3,
175        _ => {
176            error!("Unsupported MQTT version: {}", mqtt_version);
177            return Err("Unsupported MQTT version".into());
178        }
179    };
180
181    // Send initial session control message to establish the session
182    let session_control: mqtt_unified_pb::SessionControl = mqtt_unified_pb::SessionControl {
183        control_type: control_type as i32,
184        client_id: client_id.clone(),
185    };
186    let session_control_msg = MqttStreamMessage {
187        sequence_id: 0,
188        direction: mqtt_unified_pb::MessageDirection::ClientToBroker as i32,
189        payload: Some(
190            mqtt_unified_pb::mqtt_stream_message::Payload::SessionControl(session_control),
191        ),
192    };
193    gstream_sender.send(session_control_msg).await?;
194
195    // Send initial Connect message based on the version
196    let connect_packet = match connect {
197        MqttConnectPacket::V5(v5_connect) => MqttPacket::Connect5(v5_connect),
198        MqttConnectPacket::V3(v3_connect) => MqttPacket::Connect3(v3_connect),
199    };
200
201    let msg = convert_mqtt_to_stream_message(
202        &connect_packet,
203        0,
204        mqtt_unified_pb::MessageDirection::ClientToBroker,
205    )
206    .ok_or("Failed to convert connect packet to stream message")?;
207
208    gstream_sender.send(msg).await?;
209
210    // Store connection for management
211    let streaming_conn = StreamingClientConnection {
212        session_id: session_id.clone(),
213        client_id: client_id.clone(),
214        mqtt_version,
215        write_h: out_write_half,
216        read_h: in_read_half,
217        grpc_stream_sender: gstream_sender, // Unified sender for both V3 and V5
218    };
219
220    // Start the message handling loops (ConnAck will be handled in the loop)
221    start_streaming_loop(streaming_conn, inbound_gstream, parser, state).await?;
222
223    Ok(())
224}

226async fn start_streaming_loop(
227    mut conn: StreamingClientConnection,
228    mut inbound_stream: Streaming<MqttStreamMessage>,
229    mut parser: MqttParser,
230    _state: Arc<RProxyState>,
231) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
232    let sequence_counter: Arc<AtomicU64> = Arc::new(AtomicU64::new(1));
233
234    'outer: loop {
235        tokio::select! {
236            // Handle MQTT messages, client facing.
237            _ = conn.read_h.read_buf(parser.buffer_mut()) => {
238                // Parse all available packets from the buffer
239                loop {
240                    match parser.next_packet() {
241                        Ok(Some(packet)) => {
242                            debug!("Parsed MQTT packet from client {}: {:?}", conn.client_id, packet);
243                            // Convert MQTT packet to gRPC stream message using unified approach
244                            // Note: session_id is sent via gRPC metadata, not in the message
245                            let stream_msg = convert_mqtt_to_stream_message(
246                                &packet,
247                                sequence_counter.fetch_add(1, Ordering::SeqCst),
248                                mqtt_unified_pb::MessageDirection::ClientToBroker,
249                            );
250
251
252                            if let Some(msg) = stream_msg {
253                                if let Err(e) = conn.grpc_stream_sender.send(msg).await {
254                                    error!("Failed to send MQTT packet to gRPC stream for client {}: {}", conn.client_id, e);
255                                    break;
256                                } else {
257                                    debug!("MQTT packet forwarded to s-proxy for client {} (session: {}): {:?}",
258                                           conn.client_id, conn.session_id, packet);
259                                }
260                            } else {
261                                error!("Failed to convert MQTT packet to stream message for client {}: {:?}", conn.client_id, packet);
262                            }
263
264                            // Continue parsing more packets from the same buffer
265                            if parser.buffer_mut().is_empty() {
266                                break; // No more data to parse, exit inner loop
267                            }
268                        }
269                        Ok(None) => {
270                            // No more complete packets available, exit parsing loop
271                            break;
272                        }
273                        Err(ParseError::BufferEmpty) => {
274                            // No data to read, exit parsing loop
275                            debug!("No data to read for client {}, ending...", conn.client_id);
276                            break 'outer;
277                        }
278                        Err(e) => {
279                            error!("Error parsing MQTT packet from client {}: {}", conn.client_id, e);
280                            break 'outer;
281                        }
282                    }
283                }
284            }
285
286            // Handle gRPC messages, s-proxy facing
287            msg = inbound_stream.next() => {
288                match msg {
289                    Some(Ok(stream_msg)) => {
290                        if let Some(packet) = convert_stream_message_to_mqtt_packet(stream_msg) {
291                            if let Ok(mqtt_bytes) = packet.to_bytes() {
292                                if let Err(e) = conn.write_h.write_all(&mqtt_bytes).await {
293                                    error!("Failed to write MQTT response to client {}: {}, grpc payload: {:?}", conn.client_id, e, packet);
294                                    break;
295                                }
296                                debug!("Forwarded gRPC message to MQTT client {}: {:?}", conn.client_id, packet);
297                            } else {
298                                error!("Failed to convert payload to MQTT bytes for client {}: {:?}", conn.client_id, packet);
299                            }
300                        } else {
301                            error!("Failed to convert stream message to MQTT packet");
302                        }
303                    }
304                    Some(Err(e)) => {
305                        error!("gRPC stream error for client {}: {}", conn.client_id, e);
306                        break;
307                    }
308                    None => {
309                        info!("gRPC stream closed for client {}", conn.client_id);
310                        break;
311                    }
312                }
313            }
314        }
315    }
316
317    info!(
318        "Streaming client loop ended for session: {} (client: {})",
319        conn.session_id, conn.client_id
320    );
321    Ok(())
322}

324async fn wait_for_mqtt_connect(
325    stream: &mut OwnedReadHalf,
326    parser: &mut MqttParser,
327) -> Result<MqttConnectPacket, Status> {
328    // init read hint for locate the vsn
329    let mut read_hint = 7;
330
331    loop {
332        let n = stream
333            .read_buf(parser.buffer_mut())
334            .await
335            .map_err(|e| Status::internal(e.to_string()))?;
336        if n == 0 {
337            return Err(Status::unavailable("Connection closed by the client"));
338        }
339
340        if n < read_hint {
341            read_hint -= n;
342            continue;
343        } else {
344            read_hint = 0;
345        }
346
347        match parser.set_mqtt_vsn(0) {
348            Ok(_) => {}
349            Err(ParseError::More(hint, _)) => {
350                read_hint = hint;
351                continue;
352            }
353            // @TODO: handle other errors properly, like empty buffer
354            Err(e) => {
355                error!("Failed to set MQTT version: {}", e);
356                return Err(Status::internal("Failed to set MQTT version"));
357            }
358        }
359
360        match parser.next_packet() {
361            Ok(Some(MqttPacket::Connect5(connect))) => {
362                return Ok(MqttConnectPacket::V5(connect));
363            }
364            Ok(Some(MqttPacket::Connect3(connect))) => {
365                return Ok(MqttConnectPacket::V3(connect));
366            }
367            Ok(Some(_)) => {
368                return Err(Status::invalid_argument(
369                    "Expected CONNECT packet, got something else",
370                ));
371            }
372            Ok(None) => {
373                // Incomplete packet, continue reading
374            }
375            Err(ParseError::More(hint, _)) => {
376                // Need more data, continue reading
377                read_hint = hint;
378            }
379            Err(e) => {
380                error!("Error parsing CONNECT packet: {:?}", e);
381                return Err(Status::invalid_argument("Failed to parse CONNECT packet"));
382            }
383        }
384    }
385}

387#[tokio::main]
388async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
389    tracing_subscriber::fmt::init();
390    let args: Vec<String> = env::args().collect();
391
392    if args.len() != 3 {
393        eprintln!(
394            "Usage: {} <listen_port> <destination_host>:<destination_port>",
395            args[0]
396        );
397        return Err("Invalid number of arguments".into());
398    }
399
400    let listen_port = args[1].parse::<u16>()?;
401    let destination: SocketAddr = args[2].parse()?;
402
403    run_proxy(listen_port, destination).await?;
404    Ok(())
