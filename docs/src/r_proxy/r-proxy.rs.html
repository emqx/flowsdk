41async fn run_proxy(
42    listen_port: u16,
43    destination: SocketAddr,
44) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
45    let listener = TcpListener::bind(("0.0.0.0", listen_port)).await?;
46    info!("Listening on 0.0.0.0:{}", listen_port);
47
48    let proxy_state = Arc::new(RProxyState {
49        connections: Arc::new(DashMap::new()),
50        sequence_counter: AtomicU64::new(1),
51    });
52
53    loop {
54        info!("Accepting connection...");
55        let (incoming_stream, addr) = listener.accept().await?;
56        let state = proxy_state.clone();
57        info!("Accepted connection from {}", addr);
58        spawn(async move {
59            match handle_new_streaming_tcp_conn(incoming_stream, destination, state).await {
60                Ok(_) => {
61                    info!("Client connection handled successfully");
62                }
63                Err(e) => {
64                    error!("Error handling client connection: {}", e);
65                }
66            }
67        });
68    }
69}

71async fn handle_new_streaming_tcp_conn(
72    incoming: TcpStream,
73    destination: SocketAddr,
74    state: Arc<RProxyState>,
75) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
76    // Split stream to read and write halves
77    let (mut read_half, write_half) = incoming.into_split();
78    let mut parser: MqttParser = MqttParser::new();
79
80    // Wait for MQTT Connect packet
81    // @TODO: with some timeout
82    let connect = wait_for_mqtt_connect(&mut read_half, &mut parser).await?;
83
84    let client_id: String = connect.client_id.clone();
85    let session_id: String = format!(
86        "{}-{}",
87        client_id,
88        state.sequence_counter.fetch_add(1, Ordering::SeqCst)
89    );
90
91    info!(
92        "New MQTT client connecting: {} (session: {})",
93        client_id, session_id
94    );
95
96    // Connect to s-proxy with streaming
97    let channel = tonic::transport::Channel::from_shared(format!("http://{}", destination))?
98        .keep_alive_while_idle(true)
99        .keep_alive_timeout(std::time::Duration::from_secs(30))
100        .keep_alive_while_idle(true)
101        .connect()
102        .await?;
103
104    let mut grpc_client: MqttRelayServiceClient<tonic::transport::Channel> =
105        MqttRelayServiceClient::new(channel);
106
107    // Create bidirectional streaming channels
108    let (grpc_stream_sender, grpc_stream_receiver) = mpsc::channel(1000);
109
110    // Convert MQTT Connect to protobuf Connect and send directly
111    let connect_msg: mqttv5pb::MqttStreamMessage = mqttv5pb::MqttStreamMessage {
112        session_id: session_id.clone(),
113        sequence_id: 0,
114        direction: mqttv5pb::MessageDirection::ClientToBroker as i32,
115        payload: Some(mqttv5pb::mqtt_stream_message::Payload::Connect(
116            connect.into(), // Direct conversion from MQTT Connect to protobuf Connect
117        )),
118    };
119
120    grpc_stream_sender.send(connect_msg).await?;
121
122    // Establish gRPC streaming connection
123    let outbound_gstream = ReceiverStream::new(grpc_stream_receiver);
124    let request = Request::new(outbound_gstream);
125    let inbound_gstream = grpc_client
126        .stream_mqtt_messages(request)
127        .await?
128        .into_inner();
129
130    // Store connection for management
131    let streaming_conn = StreamingClientConnection {
132        session_id: session_id.clone(),
133        client_id: client_id.clone(),
134        write_h: write_half,
135        read_h: read_half,
136        grpc_stream_sender,
137    };
138
139    // Start the message handling loops (ConnAck will be handled in the loop)
140    start_streaming_client_loop(streaming_conn, inbound_gstream, state).await?;
141
142    Ok(())
143}

145async fn start_streaming_client_loop(
146    mut conn: StreamingClientConnection,
147    mut inbound_stream: Streaming<mqttv5pb::MqttStreamMessage>,
148    _state: Arc<RProxyState>,
149) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
150    let mut parser = MqttParser::new();
151    let sequence_counter = Arc::new(AtomicU64::new(1));
152
153    loop {
154        tokio::select! {
155            // Handle MQTT messages, client facing.
156            _ = conn.read_h.read_buf(parser.buffer_mut()) => {
157                match parser.next_packet() {
158                    Ok(Some(packet))=> {
159                        debug!("Parsed MQTT packet from client {}: {:?}", conn.client_id, packet);
160                        // Convert MQTT packet to gRPC stream message
161                        if let Some(payload) = convert_mqtt_to_stream_payload(&packet) {
162                            let stream_msg = mqttv5pb::MqttStreamMessage {
163                                session_id: conn.session_id.clone(),
164                                sequence_id: sequence_counter.fetch_add(1, Ordering::SeqCst),
165                                direction: mqttv5pb::MessageDirection::ClientToBroker as i32,
166                                payload: Some(payload),
167                            };
168
169                            if let Err(e) = conn.grpc_stream_sender.send(stream_msg).await {
170                                error!("Failed to send MQTT packet to gRPC stream for client {}: {}", conn.client_id, e);
171                                break;
172                            } else {
173                                debug!("MQTT packet forwarded to s-proxy for client {}", conn.client_id);
174                            }
175                        }
176                    }
177                    Ok(None) => {
178                        // Incomplete packet, continue
179                        // @TODO: handle ParseOK::Continue and use the hint to read more data
180                    }
181                    Err(ParseError::BufferEmpty) => {
182                        // No data to read, continue waiting
183                        break;
184                    }
185                    Err(e) => {
186                        error!("Error parsing MQTT packet from client {}: {}", conn.client_id, e);
187                        break;
188                    }
189                }
190            }
191
192            // Handle gRPC messages, s-proxy facing
193            msg = inbound_stream.next() => {
194                match msg {
195                    Some(Ok(stream_msg)) => {
196                        if let Some(payload) = &stream_msg.payload {
197                            // Convert all payloads (including ConnAck) to MQTT bytes and forward
198                            if let Some(mqtt_bytes) = convert_stream_payload_to_mqtt_bytes(payload) {
199                                if let Err(e) = conn.write_h.write_all(&mqtt_bytes).await {
200                                    error!("Failed to write MQTT response to client {}: {}, grpc payload: {:?}", conn.client_id, e, payload);
201                                    break;
202                                }
203                                debug!("Forwarded gRPC message to MQTT client {}: {:?}", conn.client_id, payload);
204                            } else {
205                                error!("Failed to convert payload to MQTT bytes for client {}: {:?}", conn.client_id, payload);
206                            }
207                        }
208                    }
209                    Some(Err(e)) => {
210                        error!("gRPC stream error for client {}: {}", conn.client_id, e);
211                        break;
212                    }
213                    None => {
214                        info!("gRPC stream closed for client {}", conn.client_id);
215                        break;
216                    }
217                }
218            }
219        }
220    }
221
222    info!(
223        "Streaming client loop ended for session: {} (client: {})",
224        conn.session_id, conn.client_id
225    );
226    Ok(())
227}

229async fn wait_for_mqtt_connect(
230    stream: &mut OwnedReadHalf,
231    parser: &mut MqttParser,
232) -> Result<MqttConnect, Status> {
233    loop {
234        let n = stream
235            .read_buf(parser.buffer_mut())
236            .await
237            .map_err(|e| Status::internal(e.to_string()))?;
238        if n == 0 {
239            return Err(Status::unavailable("Connection closed by the client"));
240        }
241
242        match parser.next_packet() {
243            Ok(Some(MqttPacket::Connect(connect))) => {
244                return Ok(connect);
245            }
246            Ok(Some(_)) => {
247                return Err(Status::invalid_argument(
248                    "Expected CONNECT packet, got something else",
249                ));
250            }
251            Ok(None) => {
252                // Incomplete packet, continue reading
253            }
254            Err(e) => {
255                error!("Error parsing CONNECT packet: {:?}", e);
256                return Err(Status::invalid_argument("Failed to parse CONNECT packet"));
257            }
258        }
259    }
260}

262#[tokio::main]
263async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
264    tracing_subscriber::fmt::init();
265    let args: Vec<String> = env::args().collect();
266
267    if args.len() != 3 {
268        eprintln!(
269            "Usage: {} <listen_port> <destination_host>:<destination_port>",
270            args[0]
271        );
272        return Err("Invalid number of arguments".into());
273    }
274
275    let listen_port = args[1].parse::<u16>()?;
276    let destination: SocketAddr = args[2].parse()?;
277
278    run_proxy(listen_port, destination).await?;
279    Ok(())
