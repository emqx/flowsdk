47    async fn mqtt_connect(
48        &self,
49        request: Request<mqttv5pb::Connect>,
50    ) -> Result<Response<mqttv5pb::Connack>, Status> {
51        println!("Got a request: {:?}", request);
52
53        let clientid = request.get_ref().client_id.clone();
54        let (tx, rx) = mpsc::channel(32);
55
56        let stream = match mqtt_connect_to_broker(request).await {
57            Ok(s) => s,
58            Err(e) => {
59                return Err(e);
60            }
61        };
62
63        // Create gRPC client connection to r-proxy
64        let channel = match Channel::from_static("http://[::1]:50516").connect().await {
65            Ok(channel) => channel,
66            Err(e) => {
67                eprintln!("Failed to connect to r-proxy: {}", e);
68                return Err(Status::internal("Failed to connect to r-proxy"));
69            }
70        };
71        let grpc_client = MqttRelayServiceClient::new(channel);
72
73        self.connections
74            .insert(clientid.clone(), (tx.clone(), grpc_client.clone()));
75
76        let _join = tokio::task::spawn(async move {
77            mqtt_client_loop(rx, stream, grpc_client).await;
78        });
79
80        // Return a default successful CONNACK - the real CONNACK will be forwarded by mqtt_client_loop
81        let m = mqttv5pb::Connack {
82            session_present: false,
83            reason_code: 0, // Success
84            properties: vec![],
85        };
86
87        Ok(Response::new(m))
88    }

90    async fn mqtt_publish_qos1(
91        &self,
92        request: Request<mqttv5pb::Publish>,
93    ) -> Result<Response<mqttv5pb::Puback>, Status> {
94        println!("Got a request: {:?}", request);
95        let msg_id = request.get_ref().message_id;
96
97        let req_meta = request.metadata().get("x-client-id");
98        if let Some(client_id) = req_meta.and_then(|v| v.to_str().ok()) {
99            println!("Client ID: {}", client_id);
100
101            if let Some(connection) = self.connections.get(client_id) {
102                let (tx, _) = connection.value();
103                if tx.send(request.into_inner().clone()).await.is_ok() {
104                    println!("Publish message sent successfully to proxy channel.");
105                } else {
106                    eprintln!("Failed to send publish message to proxy channel, receiver dropped.");
107                    return Err(Status::internal(
108                        "Failed to send publish message, client task may have died.",
109                    ));
110                }
111            } else {
112                return Err(Status::not_found(format!(
113                    "Client ID '{}' not found or client not connected.",
114                    client_id
115                )));
116            }
117        } else {
118            // @FIXME: x-client-id header is NOT required, but we need to handle it gracefully
119            return Err(Status::invalid_argument("Header 'x-client-id' is required"));
120        }
121
122        let m = mqttv5pb::Puback {
123            message_id: msg_id,
124            reason_code: 0,
125            properties: Default::default(),
126        };
127        Ok(Response::new(m))
128    }

312async fn mqtt_connect_to_broker(
313    request: Request<mqttv5pb::Connect>,
314) -> Result<tokio::net::TcpStream, Status> {
315    let msg: mqtt_serde::mqttv5::connect::MqttConnect = request.into_inner().into();
316    let addr = "127.0.0.1:1883".parse().unwrap();
317    let socket = TcpSocket::new_v4()?;
318    socket.set_reuseaddr(true)?;
319    let mut stream = socket.connect(addr).await?;
320    let data = msg.to_bytes().unwrap(); //@FIXME unwrap
321    println!("Connecting to MQTT broker at {} with data {:?}", addr, data);
322    stream
323        .write_all(&data)
324        .await
325        .map_err(|e| Status::internal(e.to_string()))?;
326    println!("CONNECT packet sent to MQTT broker at {}", addr);
327    Ok(stream)
328}

343async fn mqtt_client_loop(
344    mut rx: mpsc::Receiver<mqttv5pb::Publish>,
345    mut stream: tokio::net::TcpStream,
346    mut grpc_client: MqttRelayServiceClient<Channel>,
347) {
348    let mut parser = MqttParser::new();
349
350    loop {
351        tokio::select! {
352            Some(pbdata) = rx.recv() => {
353                println!("Received pb data: {:?}", pbdata);
354
355                let publish_packet: mqtt_serde::mqttv5::publish::MqttPublish = pbdata.into();
356                let data = publish_packet.to_bytes().unwrap(); //@FIXME unwrap
357                if let Err(e) = stream.write_all(&data).await {
358                    eprintln!("Failed to write to broker: {}", e);
359                    break;
360                }
361            }
362            result = stream.read_buf(parser.buffer_mut()) => {
363                match result {
364                    Ok(0) => {
365                        println!("Connection closed by the broker");
366                        break;
367                    }
368                    Ok(n) => {
369                        println!("Read {} bytes from broker", n);
370                        while let Ok(Some(packet)) = parser.next_packet() {
371                            println!("Received packet from broker: {:?}", packet);
372
373                            // Convert MQTT packet to gRPC and forward to r-proxy
374                            if let Err(e) = forward_mqtt_packet_to_grpc(&packet, &mut grpc_client).await {
375                                eprintln!("Failed to forward packet to r-proxy: {}", e);
376                            }
377                        }
378                    }
379                    Err(e) => {
380                        eprintln!("Error reading from stream: {}", e);
381                        break;
382                    }
383                }
384            }
385            else => {
386                break;
387            }
388        }
389    }
390}

392async fn forward_mqtt_packet_to_grpc(
393    packet: &InternalMqttPacket,
394    grpc_client: &mut MqttRelayServiceClient<Channel>,
395) -> Result<(), Box<dyn std::error::Error>> {
396    match packet {
397        InternalMqttPacket::Publish(publish) => {
398            let req = Request::new(publish.clone().into());
399            grpc_client.mqtt_publish_qos1(req).await?;
400        }
401        InternalMqttPacket::PubAck(puback) => {
402            let req = Request::new(puback.clone().into());
403            grpc_client.mqtt_puback(req).await?;
404        }
405        InternalMqttPacket::PubRec(pubrec) => {
406            let req = Request::new(pubrec.clone().into());
407            grpc_client.mqtt_pubrec(req).await?;
408        }
409        InternalMqttPacket::PubRel(pubrel) => {
410            let req = Request::new(pubrel.clone().into());
411            grpc_client.mqtt_pubrel(req).await?;
412        }
413        InternalMqttPacket::PubComp(pubcomp) => {
414            let req = Request::new(pubcomp.clone().into());
415            grpc_client.mqtt_pubcomp(req).await?;
416        }
417        InternalMqttPacket::SubAck(suback) => {
418            let req = Request::new(suback.clone().into());
419            grpc_client.mqtt_suback(req).await?;
420        }
421        InternalMqttPacket::UnsubAck(unsuback) => {
422            let req = Request::new(unsuback.clone().into());
423            grpc_client.mqtt_unsuback(req).await?;
424        }
425        InternalMqttPacket::PingResp(_pingresp) => {
426            // Convert PingResp to PingReq for gRPC call since broker sends PingResp
427            // but client expects PingReq pattern in reverse direction
428            let pingreq = mqttv5pb::Pingreq {};
429            let req = Request::new(pingreq);
430            grpc_client.mqtt_pingreq(req).await?;
431        }
432        InternalMqttPacket::ConnAck(connack) => {
433            // CONNACK from broker indicates successful connection
434            println!("Received CONNACK from broker: {:?}", connack);
435        }
436        InternalMqttPacket::Disconnect(disconnect) => {
437            let req = Request::new(disconnect.clone().into());
438            grpc_client.mqtt_disconnect(req).await?;
439        }
440        InternalMqttPacket::Auth(auth) => {
441            let req = Request::new(auth.clone().into());
442            grpc_client.mqtt_auth(req).await?;
443        }
444        _ => {
445            println!("Unhandled packet type from broker: {:?}", packet);
446        }
447    }
448
449    Ok(())
