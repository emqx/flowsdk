21// Control message enum for broker communication
22#[derive(Debug, Clone)]
23pub enum BrokerControlMessage {
24    Packet(flowsdk::mqtt_serde::control_packet::MqttPacket),
25}
26
27type ConnectionInfo = (
28    mpsc::Sender<BrokerControlMessage>, // Control message sender for all MQTT commands
29    mpsc::Sender<Result<MqttStreamMessage, Status>>, // Response stream sender
30    String,                             // session_id
31);

34pub struct MyRelay {

49#[tonic::async_trait]
50impl MqttRelayService for MyRelay {
51    // Streaming implementation for bidirectional communication
52    type StreamMqttMessagesStream =
53        tokio_stream::wrappers::ReceiverStream<Result<MqttStreamMessage, Status>>;
54
55    async fn stream_mqtt_messages(
56        &self,
57        request: Request<tonic::Streaming<MqttStreamMessage>>,
58    ) -> Result<Response<Self::StreamMqttMessagesStream>, Status> {
59        // Extract metadata (HTTP/2 headers) from the request
60        let metadata = request.metadata();
61        let session_id_from_metadata = metadata
62            .get("session-id")
63            .and_then(|v| v.to_str().ok())
64            .map(|s| s.to_string())
65            .unwrap_or_else(|| "unknown".to_string());
66        let client_id = metadata
67            .get("client-id")
68            .and_then(|v| v.to_str().ok())
69            .unwrap_or("unknown");
70
71        info!(
72            "Initiating bidirectional MQTT stream for session: {}, client: {}",
73            session_id_from_metadata, client_id
74        );
75
76        let mut incoming_stream = request.into_inner();
77        let (response_tx, response_rx) = tokio::sync::mpsc::channel(128);
78        let connections = self.connections.clone();
79        let broker_addr = self.broker_addr; // Capture broker_addr before move
80
81        let token = CancellationToken::new();
82
83        // Handle the bidirectional stream
84        tokio::spawn(async move {
85            let session_id = session_id_from_metadata; // Use session_id from metadata (owned)
86            loop {
87                let cloned_token: CancellationToken = token.clone();
88
89                match incoming_stream.next().await {
90                    None => {
91                        debug!("Stream closed by client, exiting grpc stream handler");
92                        token.cancel();
93                        break;
94                    }
95                    Some(Err(e)) => {
96                        error!("Error in incoming grpc stream: {}", e);
97                        break;
98                    }
99
100                    Some(Ok(stream_msg)) => {
101                        debug!("Received stream message for session: {}", &session_id);
102
103                        if let Some(payload) = stream_msg.payload {
104                            match payload {
105                                mqtt_unified_pb::mqtt_stream_message::Payload::MqttV3Packet(
106                                    packet,
107                                ) => {
108                                    if let Some(p) = packet.packet {
109                                        if let Some(client_id) =
110                                            find_client_by_session(&connections, &session_id)
111                                        {
112                                            if let Some(conn) = connections.get(&client_id) {
113                                                let (broker_tx, _, _) = conn.value();
114                                                if let Err(e) = broker_tx
115                                                    .send(BrokerControlMessage::Packet(p.into()))
116                                                    .await
117                                                {
118                                                    error!(
119                                                        "Failed to forward packet to broker for client {}: {}",
120                                                        client_id, e
121                                                    );
122                                                }
123                                            }
124                                        }
125                                    }
126                                }
127                                mqtt_unified_pb::mqtt_stream_message::Payload::MqttV5Packet(
128                                    packet,
129                                ) => {
130                                    if let Some(p) = packet.packet {
131                                        if let Some(client_id) =
132                                            find_client_by_session(&connections, &session_id)
133                                        {
134                                            if let Some(conn) = connections.get(&client_id) {
135                                                let (broker_tx, _, _) = conn.value();
136                                                if let Err(e) = broker_tx
137                                                    .send(BrokerControlMessage::Packet(p.into()))
138                                                    .await
139                                                {
140                                                    error!(
141                                                        "Failed to forward packet to broker for client {}: {}",
142                                                        client_id, e
143                                                    );
144                                                }
145                                            }
146                                        }
147                                    }
148                                }
149                                mqtt_unified_pb::mqtt_stream_message::Payload::SessionControl(
150                                    control,
151                                ) => {
152                                    // Handle session control messages
153                                    match control.control_type {
154                                        x if (x == mqtt_unified_pb::session_control::ControlType::Establishv3 as i32)
155                                                || (x == mqtt_unified_pb::session_control::ControlType::Establishv5 as i32) => {
156                                            let client_id = control.client_id.clone();
157                                            let mqtt_version = match x {
158                                                y if y == mqtt_unified_pb::session_control::ControlType::Establishv3 as i32 => 3,
159                                                y if y == mqtt_unified_pb::session_control::ControlType::Establishv5 as i32 => 5,
160                                                _ => 0, //@TODO: handle error
161                                            };
162                                            info!("Received Establish from client: {}", client_id);
163
164                                            // Create broker connection for this client
165                                            let connections = connections.clone();
166                                            let response_tx_for_spawn = response_tx.clone();
167                                            let session_id_for_spawn = session_id.clone();
168                                                       let (broker_tx, broker_rx) =
169                                                            tokio::sync::mpsc::channel::<BrokerControlMessage>(
170                                                                32,
171                                                            );
172                                                      // Store connection info
173                                                        connections.insert(
174                                                            client_id.clone(),
175                                                            (
176                                                                broker_tx,
177                                                                response_tx_for_spawn.clone(),
178                                                                session_id_for_spawn.clone(),
179                                                            ),
180                                                        );
181
182                                            tokio::spawn(async move {
183                                                match mqtt_connect_to_broker(broker_addr).await {
184                                                    Ok(stream) => {
185                                                        info!(
186                                                            "Broker connection established for client: {}",
187                                                            client_id
188                                                        );
189
190                                                        // Start the MQTT client loop to handle broker communication
191                                                        mqtt_client_loop_simple(
192                                                            cloned_token,
193                                                            connections.clone(),
194                                                            broker_rx,
195                                                            stream,
196                                                            response_tx_for_spawn,
197                                                            session_id_for_spawn,
198                                                            mqtt_version,
199                                                        )
200                                                        .await;
201                                                    }
202                                                    Err(e) => {
203                                                        error!(
204                                                            "Failed to connect to broker for client {}: {}",
205                                                            client_id, e
206                                                        );
207                                                    }
208                                                }
209                                            });
210                                        }
211                                        x if x == mqtt_unified_pb::session_control::ControlType::Terminate as i32 => {
212                                            if let Some(client_id) =
213                                                find_client_by_session(&connections, &session_id)
214                                            {
215                                                debug!("Client {} disconnecting", client_id);
216                                                connections.remove(&client_id);
217                                            }
218                                        }
219                                        _ => {}
220                                    }
221                                }
222                            }
223                        }
224                    }
225                }
226            }
227        });
228
229        // @FIXME: This is a temporary solution to return a stream
230        Ok(Response::new(tokio_stream::wrappers::ReceiverStream::new(
231            response_rx,
232        )))
233    }
234}

236async fn mqtt_connect_to_broker(broker_addr: SocketAddr) -> Result<tokio::net::TcpStream, Status> {
237    let socket = TcpSocket::new_v4()?;
238    socket.set_reuseaddr(true)?;
239    let stream = socket.connect(broker_addr).await?;
240    info!("Connected to MQTT broker at {}", broker_addr);
241    Ok(stream)
242}

244async fn mqtt_client_loop_simple(
245    token: CancellationToken,
246    connections: Arc<DashMap<String, ConnectionInfo>>,
247    mut rx: mpsc::Receiver<BrokerControlMessage>,
248    mut stream: tokio::net::TcpStream,
249    response_tx: mpsc::Sender<Result<MqttStreamMessage, Status>>,
250    session_id: String,
251    mqtt_version: u8,
252) {
253    let mut parser = MqttParser::new(16384, mqtt_version); // 16KB buffer
254    let mut sequence_counter = 1u64;
255
256    loop {
257        tokio::select! {
258            _ = token.cancelled() => {
259                // The token was cancelled
260                info!("Cancellation token triggered, exiting MQTT client loop for session: {}", session_id);
261                connections.remove(&session_id);
262                break;
263            }
264
265            // Handle control messages from r-proxy
266            Some(control_msg) = rx.recv() => {
267                debug!("Processing control message for broker");
268
269                // Convert control message to MQTT bytes and send to broker
270                let mqtt_bytes = match control_msg {
271                    BrokerControlMessage::Packet(packet) => {
272
273                        packet.to_bytes().unwrap()
274                    }
275                };
276
277                if let Err(e) = stream.write_all(&mqtt_bytes).await {
278                    error!("Failed to write to broker: {}", e);
279                    break;
280                }
281            }
282            // Handle incoming data from the broker
283            result = stream.read_buf(parser.buffer_mut()) => {
284                match result {
285                    Ok(0) => {
286                        info!("Connection closed by the broker");
287                        connections.remove(&session_id);
288                        break;
289                    }
290                    Ok(n) => {
291                        debug!("Read {} bytes from broker", n);
292                        while let Ok(Some(packet)) = parser.next_packet() {
293                            debug!("Received packet from broker: {:?}", packet);
294
295                            // Convert MQTT packet to gRPC stream message and forward to r-proxy
296                            // Note: session_id is sent via gRPC metadata, not in the message
297                            if let Some(stream_msg) = convert_mqtt_to_stream_message(
298                                &packet,
299                                sequence_counter,
300                                mqtt_unified_pb::MessageDirection::BrokerToClient,
301                            ) {
302                                sequence_counter += 1;
303
304                                if let Err(e) = response_tx.send(Ok(stream_msg)).await {
305                                    error!("Failed to send message to r-proxy via stream: {}", e);
306                                    break;
307                                }
308                            }
309
310                            debug!("Processed packet from broker, continuing to next packet");
311                        }
312                    }
313                    Err(e) => {
314                        error!("Error reading from stream: {}", e);
315                        break;
316                    }
317                }
318            }
319            else => {
320                info!("No more data to read from broker, exiting MQTT client loop for session: {}", session_id);
321                break;
322            }
323        }
324    }
325}

327// Helper function to find client by session ID
328fn find_client_by_session(
329    connections: &Arc<DashMap<String, ConnectionInfo>>,
330    session_id: &str,
331) -> Option<String> {
332    for entry in connections.iter() {
333        let (_, _, stored_session_id) = entry.value();
334        if stored_session_id == session_id {
335            return Some(entry.key().clone());
336        }
337    }
338    None
339}

341#[tokio::main]
342async fn main() -> Result<(), Box<dyn std::error::Error>> {
343    tracing_subscriber::fmt::init();
344
345    let args: Vec<String> = env::args().collect();
346
347    if args.len() != 3 {
348        error!(
349            "Usage: {} <grpc_listen_port> <broker_host>:<broker_port>",
350            args[0]
351        );
352        error!("Example: {} 50515 127.0.0.1:1883", args[0]);
353        return Err("Invalid number of arguments".into());
354    }
355
356    let grpc_listen_port = args[1].parse::<u16>()?;
357    let broker_addr: SocketAddr = args[2].parse()?;
358
359    let grpc_addr = format!("127.0.0.1:{}", grpc_listen_port).parse()?;
360    let relay = MyRelay::new(broker_addr);
361
362    info!("s-proxy starting:");
363    info!("  gRPC server listening on: {}", grpc_addr);
364    info!("  MQTT broker address: {}", broker_addr);
365
366    Server::builder()
367        .add_service(MqttRelayServiceServer::new(relay))
368        .serve(grpc_addr)
369        .await?;
370
371    Ok(())
